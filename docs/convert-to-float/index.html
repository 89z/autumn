Go

package main
import "fmt"
func main() {
   n1 := float64(0)
   n2 := float64(0)
   // example 1
   fmt.Sscanln("1.8 1.9", &n1, &n2)
   println(n1, n2)
   // example 2
   fmt.Sscan("1.8\n1.9", &n1, &n2)
   println(n1, n2)
   // example 3
   fmt.Sscanf("1.8 1.9", "%v %v", &n1, &n2)
   println(n1, n2)
}

package main
func main() {
   n1 := 10
   n2 := float64(n1)
   println(n2)
}

package main
import (
   "os"
   "strconv"
)
func main() {
   s := "1.9"
   n, e := strconv.ParseFloat(s, 64)
   if e != nil {
      os.Exit(1)
   }
   println(n)
}

Rust

use std::num;
fn main() -> Result<(), num::ParseFloatError> {
   let s = "1.9";
   let n: f32 = s.parse()?;
   println!("{}", n == 1.9);
   Ok(())
}
