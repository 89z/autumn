Fri Oct 2 2020 number 

package main
import "fmt"

func main() {
   // example 1
   n1 := float64(0)
   fmt.Sscanln("9.9", &n1)
   // example 2
   n2 := float64(0)
   fmt.Sscan("9.9", &n2)
   // example 3
   n3 := float64(0)
   fmt.Sscanf("9.9", "%v", &n3)
   // print
   println(n1 == 9.9, n2 == 9.9, n3 == 9.9)
}

package main

func main() {
   n := 10
   n1 := float64(n)
   println(n1 == 10, n1 == 10.0)
}

package main

import (
   "os"
   "strconv"
)

func main() {
   s := "9.9"
   n, e := strconv.ParseFloat(s, 64)
   if e != nil {
      os.Exit(1)
   }
   println(n == 9.9)
}
