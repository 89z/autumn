String to float / PHP

Fri Oct 2 2020 number 

<?php
$s = '9.9';
# example 1
$n1 = +($s);
# example 2
$n2 = (float)($s);
# example 3
$n3 = floatval($s);
# print
var_dump($n1 === 9.9, $n2 === 9.9, $n3 === 9.9);

<?php
$n = 10;
# example 1
$n1 = (float)($n);
# example 2
$n2 = floatval($n);
# print
var_dump($n1 === 10.0, $n2 === 10.0);
