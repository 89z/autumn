Date string to object

Tue Jul 14 2020 date

D

import std.datetime, std.stdio;
void main() {
   auto s = "2019-12-31";
   auto o = Date.fromISOExtString(s);
   writeln(o);
}

Dart

main() {
   var s = '2019-12-31';
   var o = DateTime.parse(s);
   print(o);
}

https://api.dart.dev/dart-core/DateTime/parse.html

Go

package main
import (
   "fmt"
   "log"
   "time"
)
func main() {
   s := time.RFC3339[:10]
   o, e := time.Parse(s, "2019-12-31")
   if e != nil {
      log.Fatal(e)
   }
   fmt.Println(o)
}

https://golang.org/pkg/time#Parse

Python

from datetime import date
o = date.fromisoformat('2019-12-31')
print(o)

from datetime import datetime
s = '2019-12-31'
# example 1
o1 = datetime.strptime(s, '%Y-%m-%d')
# example 2
o2 = datetime.fromisoformat(s)
# print
print(o1, o2)

import time
o = time.strptime('2019-12-31', '%Y-%m-%d')
print(o)
