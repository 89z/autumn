Dart

import 'dart:io';
main() {
   Process.run('less', ['-V']).then((ProcessResult results) {
      stdout.write(results.stdout);
   });
}

https://api.dart.dev/dart-io/Process-class.html

Go

package main
import (
   "fmt"
   "log"
   "os/exec"
)
func main() {
   o := exec.Command("ag", "-V")
   y, e := o.Output()
   if e != nil {
      log.Fatal(e)
   }
   fmt.Printf("%s", y)
}

package main
import (
   "os/exec"
   "strings"
)
func main() {
   var o1 = exec.Command("ag")
   var o2 strings.Builder
   o1.Stdout = &o2
   o1.Run()
   var s1 = o2.String()
   print(s1)
}

  • https://golang.org/pkg/strings

Python

import subprocess
a1 = ['ag', '-V']
# example 1
o1 = subprocess.Popen(a1, stdout=subprocess.PIPE, text=True)
o2 = o1.stdout
s1 = o2.read()
# example 2
o3 = subprocess.run(a1, stdout=subprocess.PIPE, text=True)
s2 = o3.stdout
# example 3
s3 = subprocess.check_output(a1, text=True)
# print
print(s1, s2, s3, end='', sep='')
  • https://docs.python.org/library/subprocess.html#subprocess.Popen
