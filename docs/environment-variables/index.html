Environment variables

Fri Sep 11 2020 system

C#

using System;

class Program {
   static void Main() {
      var s = "USERPROFILE";
      var s1 = Environment.GetEnvironmentVariable(s);
      Console.WriteLine(s1 == @"C:\Users\Steven");
   }
}

Go

package main
import "os"

func main() {
   s := "USERPROFILE"
   // example 1
   s1 := os.Getenv(s)
   // example 2
   s2, b2 := os.LookupEnv(s)
   // print
   s0 := `C:\Users\Steven`
   println(s1 == s0, s2 == s0, b2)
}

Rust

use std::env;

fn main() -> Result<(), env::VarError> {
   // example 1
   let s1 = env::var("USERPROFILE")?;
   // example 2
   let s2 = env!("USERPROFILE");
   // print
   println!("{}", s1 == r"C:\Users\Steven" && s2 == r"C:\Users\Steven");
   Ok(())
}
