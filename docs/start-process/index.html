Start process

Fri May 15 2020 system Talk:Start process

Go

package main
import (
   "os"
   "os/exec"
)
func main() {
   o1 := exec.Command("less", "-V")
   o1.Stdout = os.Stdout
   o1.Run()
}

package main
import (
   "os"
   "os/exec"
)
func main() {
   o2 := exec.Command("less", "index.md")
   o2.Stdin = os.Stdin
   o2.Stdout = os.Stdout
   o2.Start()
}

<?php
function f_escape_sh($s_all) {
   $a_enc = explode('"', $s_all);
   $a_tran = array_map(fn($s_one) => '"' . $s_one . '"', $a_enc);
   return implode('"', $a_tran);
}
$a1 = ['less', '-V'];
$a2 = array_map('f_escape_sh', $a1);
$s1 = implode(' ', $a2);
system($s1);
