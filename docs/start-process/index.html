Dart

import 'dart:io';
main() {
   var s = Platform.environment['BROWSER'];
   Process.runSync(s, ['https://dart.dev']);
}

Go

package main
import (
   "os"
   "os/exec"
)
func main() {
   o := exec.Command("less", "-V")
   o.Stdout = os.Stdout
   o.Run()
}

package main
import (
   "os"
   "os/exec"
)
func main() {
   o := exec.Command("less", "index.md")
   o.Stdin = os.Stdin
   o.Stdout = os.Stdout
   o.Start()
}

<?php
function f_escape_sh($s_in) {
   $s_out = preg_replace('/"/', '""', $s_in);
   if ($s_out != $s_in) {
      return '"' . $s_out . '"';
   }
   $n_mat = preg_match('/[ &>^-]/', $s_in);
   if ($n_mat == 1) {
      return '"' . $s_out . '"';
   }
   return $s_out;
}
$s = f_escape_sh('a b.txt');
system('less ' . $s);
