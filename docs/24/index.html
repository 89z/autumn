Condition-first loops

D

import std.stdio;
void main() {
   auto n = 10;
   while (n < 20) {
      n.writeln;
      n++;
   }
}

Dart

main() {
   var n = 10;
   while (n < 20) {
      print(n);
      n++;
   }
}

https://dart.dev/guides/language/language-tour#while-and-do-while

Go

package main

func main() {
   println("example 1")
   for n := 10; n < 20; n++ {
      println(n)
   }
   println("example 2")
   n := 10
   for n < 20 {
      println(n)
      n++
   }
}

https://golang.org/ref/spec#For_statements

JavaScript

console.log('example 1');
for (let n = 10; n < 20; n++) {
   console.log(n);
}

console.log('example 2');
let n = 10;
while (n < 20) {
   console.log(n);
   n++;
}

   var_dump($n);

   var_dump($n);
}
