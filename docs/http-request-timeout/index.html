HTTP request timeout

Sun Apr 26 14:15:11 2020 network

PHP

<?php
$s1 = 'http://speedtest.lax.hivelocity.net:10';
$r1 = stream_context_create();
stream_context_set_option($r1, 'http', 'timeout', 5);
file_get_contents($s1, false, $r1);

<?php
$s1 = 'http://speedtest.lax.hivelocity.net:10';
$r1 = curl_init($s1);
curl_setopt($r1, CURLOPT_TIMEOUT, 5);
curl_exec($r1);

  • https://php.net/function.curl-setopt
  • https://php.net/function.stream-context-set-option

Python

import urllib.request as v1
s1 = 'http://speedtest.lax.hivelocity.net:10'
v1.urlopen(s1, timeout=5)
