Get array length

Wed May 27 2020 array

Dart

main() {
   var a = [10, 11];
   var n = a.length;
   print(n);
}

Go

package main
func main() {
   a1 := []string{"Sun", "Mon"}
   n1 := len(a1)
   println(n1)
}

https://golang.org/pkg/builtin#len

JavaScript

let a1 = ['Sun', 'Mon'];
let n1 = a1.length;
console.log(n1);

https://developer.mozilla.org/Web/JavaScript/Reference/Global_Objects/Array/length

Nim

let a1 = ["Sun", "Mon"]
let n1 = a1.len
echo n1

<?php
