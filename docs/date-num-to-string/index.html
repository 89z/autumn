Date number to string

Fri Jun 12 2020 date

D

import std.datetime, std.stdio;
void main() {
   auto n = 366 * 24 * 60 * 60;
   auto o = SysTime.fromUnixTime(n);
   auto s = o.toISOExtString;
   s.writeln;
}

https://dlang.org/library/std/datetime/systime/sys_time.from_unix_time.html

Go

package main
import "time"
func main() {
   var n int64 = 366 * 24 * 60 * 60
   var o = time.Unix(n, 0)
   var s = o.Format("Mon Jan 2 2006")
   println(s == "Fri Jan 1 1971")
}

https://golang.org/pkg/time#Time.Format

JavaScript

let n = 366 * 24 * 60 * 60;
let o = new Date(n * 1000);
let s_d = o.toLocaleString(0, {weekday: 'short'});
let s_m = o.toLocaleString(0, {month: 'short', day: 'numeric'});
let s_y = o.toLocaleString(0, {year: 'numeric'});
let s = s_d + ' ' + s_m + ' ' + s_y;
console.log(s == 'Fri Jan 1 1971');

https://developer.mozilla.org/Web/JavaScript/Reference/Global_Objects/Date/toLocaleString

PHP

<?php
$n = 365 * 24 * 60 * 60;
# example 1
$s1 = strftime('%a %b %#d %Y', $n);
# example 2
$s2 = strftime('%a %b %#e %Y', $n);
# print
var_dump($s1 == 'Fri Jan 1 1971', $s2 == 'Fri Jan 1 1971');

Python

from datetime import date
n = 366 * 24 * 60 * 60
o = date.fromtimestamp(n)
s = o.strftime('%a %b %#d %Y')
print(s == 'Fri Jan 1 1971')
