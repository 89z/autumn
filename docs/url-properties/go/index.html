package main

import (
   "log"
   "net/url"
)

func main() {
   u, e := url.Parse("https://example.com/one?two=even")
   if e != nil {
      log.Fatal(e)
   }
   v := u.Query()
   log.Print(v)
}

package main
import "net/url"

func main() {
   v := url.Values{
      "one": []string{"odd"},
      "two": []string{"even"},
   }
   // example 1
   s1 := v.Get("one")
   // example 2
   s2 := v["one"][0]
   // print
   println(s1 == "odd", s2 == "odd")
}

package main
import "net/url"

func main() {
   v := make(url.Values)
   v.Set("one", "odd")
   v.Set("two", "even")
   s := v.Encode()
   println(s == "one=odd&two=even")
}
