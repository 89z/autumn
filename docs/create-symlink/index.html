Create symlink

Fri Jul 3 2020 system Talk:Create symlink

Go

package main
import (
   "log"
   "os"
)
func main() {
   s_old, e := os.Getwd()
   if e != nil {
      log.Fatal(e)
   }
   s_new := os.Getenv("TMP")
   os.Symlink(s_old + "/index.md", s_new + "/index.md")
}

https://golang.org/pkg/os#Symlink
