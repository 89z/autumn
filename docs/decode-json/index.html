Go

package main
import (
   "encoding/json"
   "fmt"
   "io/ioutil"
)
func main() {
   a1, _ := ioutil.ReadFile("a.json")
   // example 1
   var m1 map[string]int
   json.Unmarshal(a1, &m1)
   fmt.Println(m1)
   // example 2
   var o1 struct{Sunday, Monday int}
   json.Unmarshal(a1, &o1)
   fmt.Printf("%+v\n", o1)
}

PHP

<?php
$s1 = file_get_contents('a.json');
# example 1
$o1 = json_decode($s1);
$n1 = $o1->Sunday;
# example 2
$m1 = json_decode($s1, true);
$n2 = $m1['Sunday'];
# print
var_dump($n1, $n2);

Requires php-json package.

https://php.net/function.json-decode

Python

import json
# example 1
o1 = open('a.json')
m1 = json.load(o1)
# example 2
o2 = open('a.json')
s1 = o2.read()
m2 = json.loads(s1)
# print
print(m1, m2)
