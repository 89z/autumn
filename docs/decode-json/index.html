Dart

import 'dart:convert';
main() {
   var s = '{"Sunday": 10}';
   var m = jsonDecode(s);
   print(m);
}

Go

package main
import (
   "encoding/json"
   "fmt"
)
func main() {
   var y = []byte(`{"Sunday": 10}`)
   // example 1
   var m map[string]int
   json.Unmarshal(y, &m)
   fmt.Println(m)
   // example 2
   var o struct {Sunday int}
   json.Unmarshal(y, &o)
   fmt.Printf("%+v\n", o)
}
  • https://golang.org/pkg/encoding/json#Unmarshal
  • https://golang.org/ref/spec#Exported_identifiers
