Iterate a set / JavaScript

Thu Aug 27 2020 iterate set 

let t = new Set(['May', 'June']);
// example 1
for (let s of t) {
   console.log(s);
}
// example 2
let f = s => console.log(s);
t.forEach(f);

let t = {'May': true, 'June': true};

for (let s in t) {
   console.log(s);
}
