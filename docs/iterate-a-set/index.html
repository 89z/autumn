Iterate a set

Thu Aug 27 2020 iterate set

C#

using System;
using System.Collections.Generic;
 
class Program {
   static void Main() {
      var t = new HashSet<string>{"May", "June"};
      foreach (var s in t) {
         Console.WriteLine(s);
      }
   }
}

JavaScript

let t = new Set(['May', 'June']);
// example 1
for (let s of t) {
   console.log(s);
}
// example 2
let f = s => console.log(s);
t.forEach(f);

let t = {'May': true, 'June': true};

for (let s in t) {
   console.log(s);
}

Rust

use std::collections::HashSet;

fn main() {
   let mut t = HashSet::new();
   t.insert("May");
   t.insert("June");
   for s in t {
      println!("{}", s);
   }
}
