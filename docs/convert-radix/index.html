JavaScript

function f1(ni, nr) {
   let s1 = '0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ';
   let s2 = '';
   while (ni > 0) {
      s2 = s1[ni % nr] + s2;
      ni = Math.trunc(ni / nr);
   }
   return s2;
}

Python

import string

def radix64(n_div):
   s_safe = string.digits + string.ascii_letters + '-_'
   n_safe = len(s_safe)
   s_out = ''
   while n_div > 0:
      n_div, n_mod = divmod(n_div, n_safe)
      s_out = s_safe[n_mod] + s_out
   return s_out

n1 = 1588337932
s1 = radix64(n1)
print(s1)
