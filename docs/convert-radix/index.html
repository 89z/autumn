Convert radix

Mon May 25 2020 number Talk:Convert radix

Some faster bitwise operations could be used, but these functions allow to easily change the radix. These also account for numeric zero input.

Go

package main
import (
   "math/big"
   "time"
)
// example 1
func r64_encode(n_in int64) string {
   o_big := big.NewInt(n_in)
   return o_big.Text(62)
}
// example 2
func r64_decode(s_in string) int64 {
   var o_big big.Int
   o_big.SetString(s_in, 62)
   return o_big.Int64()
}
// print
func main() {
   n1 := time.Now().Unix()
   s1 := r64_encode(n1)
   n2 := r64_decode(s1)
   println(n1, s1, n2 == n1)
}

package main
import (
   "strings"
   "time"
)
var s_dig = "-0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ_abcdefghijklmnopqrstuvwxyz"
// example 1
func r64_encode(n_in int) string {
   var s_out string
   for {
      n_key := n_in % 64
      s_out = s_dig[n_key:n_key + 1] + s_out
      n_in /= 64
      if n_in == 0 {
         break
      }
   }
   return s_out
}
// example 1
func r64_decode(s_in string) int {
   var n_out int
   for n_key := range s_in {
      s_chr := s_in[n_key:n_key + 1]
      n_out = n_out * 64 + strings.Index(s_dig, s_chr)
   }
   return n_out
}
// print
func main() {
   n1 := int(time.Now().Unix())
   s1 := r64_encode(n1)
   n2 := r64_decode(s1)
   println(n1, s1, n2 == n1)
}

JavaScript

let s_dig = '-0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ_abcdefghijklmnopqrstuvwxyz';
// example 1
function r64_encode(n_in) {
   let s_out = '';
   do {
      s_out = s_dig[n_in % 64] + s_out;
      n_in = Math.trunc(n_in / 64);
   } while (n_in > 0);
   return s_out;
}
// example 2
function r64_decode(s_in) {
   let n_out = 0;
   for (let s_chr of s_in) {
      n_out = n_out * 64 + s_dig.indexOf(s_chr);
   }
   return n_out;
}
// print
let n1 = Math.trunc(new Date / 1000);
let s1 = r64_encode(n1);
let n2 = r64_decode(s1);
console.log(n1, s1, n2 == n1);

<?php
$s_dig = '-0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ_abcdefghijklmnopqrstuvwxyz';
# example 1
function r64_encode($n_in) {
   global $s_dig;
   $s_out = '';
   do {
      $s_out = $s_dig[$n_in % 64] . $s_out;
      $n_in = intdiv($n_in, 64);
   } while ($n_in > 0);
   return $s_out;
}
# example 2
function r64_decode($s_in) {
   global $s_dig;
   $n_out = 0;
   $a_in = str_split($s_in);
   foreach ($a_in as $s_chr) {
      $n_out = $n_out * 64 + strpos($s_dig, $s_chr);
   }
   return $n_out;
}
# print
$n1 = time();
$s1 = r64_encode($n1);
$n2 = r64_decode($s1);
var_dump($n1, $s1, $n2 == $n1);
