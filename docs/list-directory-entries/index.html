Go

package main
import (
   "io/ioutil"
   "log"
)
func main() {
   a, e := ioutil.ReadDir(".")
   if e != nil {
      log.Fatal(e)
   }
   for n, o := range a {
      s := o.Name()
      println(n, s)
   }
}

package main
import (
   "log"
   "os"
)
func main() {
   f, e := os.Open(".")
   if e != nil {
      log.Fatal(e)
   }
   a, e := f.Readdirnames(0)
   if e != nil {
      log.Fatal(e)
   }
   log.Println(a)
}

package main
import (
   "fmt"
   "log"
   "path/filepath"
)
func main() {
   a, e := filepath.Glob("*")
   if e != nil {
      log.Fatal(e)
   }
   fmt.Println(a)
}
