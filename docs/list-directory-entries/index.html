D

import std.file, std.stdio;
void main() {
   auto a1 = dirEntries(".", SpanMode.shallow);
   a1.writeln;
}

https://dlang.org/phobos/std_file.html#.dirEntries

Go

package main
import "io/ioutil"
func main() {
   a1, _ := ioutil.ReadDir(".")
   for _, o1 := range a1 {
      s1 := o1.Name()
      println(s1)
   }
}

https://golang.org/pkg/io/ioutil#ReadDir

package main
import (
   "fmt"
   "os"
)
func main() {
   f1, _ := os.Open(".")
   a1, _ := f1.Readdirnames(0)
   fmt.Println(a1)
}

https://golang.org/pkg/os#File.Readdirnames
