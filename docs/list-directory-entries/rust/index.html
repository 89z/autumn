List directory entries / Rust

Sun Sep 27 2020 system 
use std::{fs, io};

fn main() -> io::Result<()> {
   for r_ent in fs::read_dir(".")? {
      let s_ent = r_ent?.path();
      println!("{:?}", s_ent);
   }
   Ok(())
}
