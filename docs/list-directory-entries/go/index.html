List directory entries / Go

Sun Sep 27 2020 system 

package main

import (
   "io/ioutil"
   "os"
)

func main() {
   a, e := ioutil.ReadDir(".")
   if e != nil {
      os.Exit(1)
   }
   for n, o := range a {
      s := o.Name()
      println(n, s)
   }
}

package main

import (
   "fmt"
   "log"
   "os"
)

func main() {
   o, e := os.Open(".")
   if e != nil {
      log.Fatal(e)
   }
   a, e := o.Readdirnames(0)
   if e != nil {
      log.Fatal(e)
   }
   fmt.Println(a)
}

package main

import (
   "log"
   "os"
   "path/filepath"
)

func f(s string, o os.FileInfo, e error) error {
   if e != nil {
      return e
   }
   if ! o.IsDir() {
      println(s)
   }
   return nil
}

func main() {
   e := filepath.Walk(".", f)
   if e != nil {
      log.Fatal(e)
   }
}
