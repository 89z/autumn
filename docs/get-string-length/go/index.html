func main() {
   s := "ðŸ“—"
   // example 1
   n1 := len(s)
   // example 2
   a := []rune(s)

   // print
   println(n1 == 4, n2 == 1)
}

package main
import "unicode/utf8"

func main() {
   s := "ðŸ“—"
   n := utf8.RuneCountInString(s)
   println(n == 1)
}
