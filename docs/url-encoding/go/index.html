package main

import (
   "fmt"
   "log"
   "net/url"
)

func main() {
   s := "http://golang.org"
   u, e := url.Parse(s)
   if e != nil {
      log.Fatal(e)
   }
   fmt.Println(u)
}

package main
import "net/url"

func main() {
   u := url.URL{Host: "golang.org"}
   s := u.String()
   println(s)
}
