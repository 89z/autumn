Iterate a map

Wed May 27 2020 iterate map

Go

package main
func main() {
   m1 := map[string]int{"Sun": 10, "Mon": 11}
   // example 1
   for s1 := range m1 {
      println(s1)
   }
   // example 2
   for s1, n1 := range m1 {
      println(s1, n1)
   }
}

https://golang.org/ref/spec#RangeClause

JavaScript

let m1 = {Sun: 10, Mon: 11};
// example 1
for (let s1 in m1) {
   console.log(s1);
}
// example 2
for (let [s1, n1] of Object.entries(m1)) {
   console.log(s1, n1);
}

PHP

<?php
$m = ['Sun' => 10, 'Mon' => 11];
# example 1
while (true) {
   $n = current($m);
   if ($n === false) {
      break;
   }
   $s = key($m);
   var_dump($s, $n);
   next($m);
}
# example 2
foreach ($m as $n) {
   var_dump($n);
}
# example 3
foreach ($m as $s => $n) {
   var_dump($s, $n);
}
  • https://php.net/control-structures.foreach
  • https://php.net/function.prev

Python

m1 = {'Sun': 10, 'Mon': 11}
for s1, n1 in m1.items():
