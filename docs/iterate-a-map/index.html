Iterate a map

Wed Mar 4 18:45:59 2020 iterate map

Go

package main
func main() {
   m1 := map[string]int{"sun": 10, "mon": 11}
   // example 1
   for s1 := range m1 {
      println(s1)
   }
   // example 2
   for s1, n1 := range m1 {
      println(s1, n1)
   }
}

https://golang.org/ref/spec#RangeClause

JavaScript

let m1 = {sun: 10, mon: 11};
// example 1
for (let s1 in m1) {
   console.log(s1);
}
// example 2
for (let [s1, n1] of Object.entries(m1)) {
   console.log(s1, n1);
}
  • https://developer.mozilla.org/Web/JavaScript/Reference/Statements/for...in
  • https://developer.mozilla.org/Web/JavaScript/Reference/Global_Objects/Object/entries

PHP

<?php
$m1 = ['sun' => 10, 'mon' => 11];
# example 1
foreach ($m1 as $n1) {
   var_dump($n1);
}
# example 2
end($m1);
while (true) {
   $s1 = key($m1);
   if ($s1 === null) {
      break;
   }
   $n1 = current($m1);
   var_dump($s1, $n1);
   prev($m1);
}
  • https://php.net/control-structures.foreach
  • https://php.net/function.prev
