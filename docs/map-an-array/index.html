Map an array

Fri Jun 12 2020 array

Dart

main() {
   var a1 = ['Monday', 'Tuesday'];
   var a2 = a1.map((s) => s.length);
   print(a2);
}

https://api.dart.dev/dart-core/Iterable/map.html

Go

package main
import "fmt"
func main() {
   a1 := []string{"Sun", "Mon"}
   for n1 := range a1 {
      a1[n1] += "day"
   }
   fmt.Println(a1)
}

https://golang.org/ref/spec#For_range

JavaScript

let a1 = [16, 25];
// example 1
let a2 = a1.map(n1 => Math.sqrt(n1));
// example 2
let a3 = a1.map(Math.sqrt);
// example 3
let a4 = Array.from(a1, n1 => Math.sqrt(n1));
// print
console.log(a2, a3, a4);
  • https://developer.mozilla.org/Web/JavaScript/Reference/Global_Objects/Array/from
  • https://developer.mozilla.org/Web/JavaScript/Reference/Global_Objects/Array/map

PHP

<?php
$a1 = ['Sun', 'Mon'];
# example 1
$a2 = array_map('strlen', $a1);
# example 2
$a3 = array_map(fn($s1) => strlen($s1), $a1);
# print
var_dump($a2, $a3);

https://php.net/function.array-map

Python

import math
a1 = [16, 25]
# example 1
a2 = [math.sqrt(n1) for n1 in a1]
# example 2
a3 = list(map(math.sqrt, a1))
# print
print(a2, a3)
  • https://docs.python.org/library/functions.html#map

Rust

fn main() {
   let a1 = [10, 11];
   let a2: Vec<u8> = a1.iter().map(|n| n * 2).collect();
   println!("{:?}", a2);
}
