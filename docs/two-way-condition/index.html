D

import std.stdio;

void main() {
   int n = 10;
   string s;
   if (n > 0) {
      s = "+";
   } else if (n < 0) {
      s = "-";
   } else {
      s = "zero";
   }
   writeln(s == "+");
}

Go

package main

func main() {
   n := 10
   // example A
   sA := ""
   if n > 0 {
      sA = "+"
   } else if n < 0 {
      sA = "-"
   } else {
      sA = "zero"
   }
   // example B
   sB := map[bool]string{true: "+", false: "-"}[n > 0]
   // print
   println(sA == "+", sB == "+")
}
  • https://golang.org/ref/spec#If_statements
  • Go ?: operator

JavaScript

let n = 10;
// example A
let sA;
if (n > 0) {
   sA = '+';
} else if (n < 0) {
   sA = '-';
} else {
   sA = 'zero';
}
// example B
let sB = n > 0 ? '+' : '-';
// print
console.log(sA == '+', sB == '+');
