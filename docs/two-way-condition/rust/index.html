Two-way condition / Rust

Fri Sep 11 2020 reference
Talk:Two-way condition 
fn main() {
   let n = 10;
   let s = if n > 0 {
      "+"
   } else if n < 0 {
      "-"
   } else {
      "zero"
   };
   println!("{}", s == "+");
}
