Parse command line arguments / Ruby

Sun Nov 8 2020 system 
require 'getoptlong'
pre_s = ''
suf_s = ''

GetoptLong.new(
   ['-p', GetoptLong::REQUIRED_ARGUMENT],
   ['-s', GetoptLong::REQUIRED_ARGUMENT]
).each { |key_s, val_s|
   case key_s
   when '-p'
      pre_s = val_s
   when '-s'
      suf_s = val_s
   end
}

if ARGV.length != 1
   puts <<eof
add [flags] <stem>
-p string
   prefix
-s string
   suffix
eof
   exit 1
end

puts pre_s + ARGV[0] + suf_s

require 'optparse'
m = {:p => '', :s => ''}
o = OptionParser.new
o.on('-p string', 'prefix')
o.on('-s string', 'suffix')
o.parse!(into: m)

if ARGV.length != 1
   o.banner = 'add [flags] <stem>'
   puts o
   exit 1
end

puts m[:p] + ARGV[0] + m[:s]
