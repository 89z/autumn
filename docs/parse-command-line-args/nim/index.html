import os, parseopt
var argCtr: int
for kind, key, value in getOpt():
   case kind
   of cmdArgument:
      echo "Got arg ", argCtr, ": \"", key, "\""
      argCtr.inc
   of cmdLongOption, cmdShortOption:
      case key
      of "v", "n", "z", "w":
         echo "Got a \"", key, "\" option with value: \"", value, "\""
      else:
      echo "Unknown option: ", key
   of cmdEnd:
      discard

import parseopt, strutils

var
   n_len = 1
   n_start: int
   o_opt = initOptParser()
   s_in: string

while true:
   o_opt.next()
   if o_opt.kind == cmdEnd:
      break
   case o_opt.key
   of "start":
      n_start = o_opt.val.parseInt
   of "len":
      n_len = o_opt.val.parseInt
   else:
      s_in = o_opt.key

let s_out = s_in[n_start ..< n_start + n_len]
echo s_out
