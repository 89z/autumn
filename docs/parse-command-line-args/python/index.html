import argparse

parser = argparse.ArgumentParser()
parser.add_argument('--count', '-c',
  dest='count', type=int)
parser.add_argument('--ratio', '-r',
  dest='ratio', type=float)
args = parser.parse_args()

if args.count:
  print('Count: %d' % args.count)
if args.ratio:
  print('Ratio: %.3f' % args.ratio)
