package main
import "fmt"

func main() {
   { // int
      var n int
      fmt.Sscan("100", &n)
      fmt.Println(n == 100)
   }
   { // float
      var n float64
      fmt.Sscan("99.9", &n)
      fmt.Println(n == 99.9)
   }
}

package main
import "strconv"

func main() {
   n, e := strconv.Atoi("100")
   if e != nil {
      panic(e)
   }
   println(n == 100)
}

package main
import "strconv"

func main() {
   { // int
      n, e := strconv.ParseInt("100", 10, 0)
      if e != nil {
         panic(e)
      }
      println(n == 100)
   }
   { // int64
      n, e := strconv.ParseInt("100", 10, 64)
      if e != nil {
         panic(e)
      }
      println(n == 100)
   }
}
