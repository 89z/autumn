Filepath to dirname

Sat Sep 5 2020 string

C++

#include <filesystem>
#include <iostream>
namespace fs = std::filesystem;
int main() {
   fs::path o = "C:\\Windows\\write.exe";
   auto s = o.parent_path();
   std::cout << s << std::endl;
}

D

import std.path, std.stdio;
void main() {
   auto s1 = `C:\Windows\write.exe`;
   auto s2 = s1.dirName;
   writeln(s2 == `C:\Windows`);
}

Go

package main
import "path"
func main() {
   s1 := "C:/go/bin/go.exe"
   s2 := path.Dir(s1)
   println(s2 == "C:/go/bin")
}

package main
import "path/filepath"
func main() {
   s1 := `C:\go\bin\go.exe`
   s2 := filepath.Dir(s1)
   println(s2 == `C:\go\bin`)
}
