Filepath to dirname

Thu Sep 10 2020 system

C#

using System;
using System.IO;

class Program {
   static void Main() {
      var s = @"C:\Windows\notepad.exe";
      var s1 = Path.GetDirectoryName(s);
      Console.WriteLine(s1 == @"C:\Windows");
   }
}

D

import std.path, std.stdio;

void main() {
   auto s = `C:\Windows\notepad.exe`;
   auto s1 = s.dirName;
   writeln(s1 == `C:\Windows`);
}

Go

package main
import "path"

func main() {
   s := "C:/go/bin/go.exe"
   s1 := path.Dir(s)
   println(s1 == "C:/go/bin")
}

package main
import "path/filepath"

func main() {
   s := `C:\Windows\notepad.exe`
   s1 := filepath.Dir(s)
   println(s1 == `C:\Windows`)
}

PHP

<?php
$s = 'C:\\Windows\\notepad.exe';
# example 1
$s1 = pathinfo($s, PATHINFO_DIRNAME);

from os import path
s = r'C:\Windows\notepad.exe'
s1 = path.dirname(s)
print(s1 == r'C:\Windows')
