Slice string from middle

Tue May 5 2020 string

Go

package main
func main() {
   s1 := "Sunday"
   // example 1
   s2 := s1[1:2]
   // example 2
   s3 := s1[1:]
   // print
   println(s2 == "u", s3 == "unday")
}

package main
func main() {
   s_in := "♠♣♥♦"
   a_dec := []rune(s_in)
   a_tran := a_dec[1:2]
   s_enc := string(a_tran)
   println(s_enc == "♣")
}

JavaScript

let s1 = 'Sunday';
let s2;
// example 1
s2 = s1.slice(1, 2);
console.log(s2 == 'u');
// example 2
s2 = s1.substring(1, 2);
console.log(s2 == 'u');
// example 3
s2 = s1.charAt(1);
console.log(s2 == 'u');
// example 4
s2 = s1[1];
console.log(s2 == 'u');
// example 5
s2 = s1.slice(1);
console.log(s2 == 'unday');
// example 6
s2 = s1.substring(1);
console.log(s2 == 'unday');
