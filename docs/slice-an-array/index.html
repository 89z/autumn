Slice an array

Wed Mar 25 10:17:37 2020 array

D

import std.stdio;
void main() {
   auto a1 = ["Sun", "Mon", "Tue"];
   // string from front
   auto s1 = a1[0];
   // string from back
   auto s2 = a1[$ - 1];
   // array from front
   auto a2 = a1[0 .. 2];
   // array from back
   auto a3 = a1[$ - 2 .. $];
   // print
   writeln(s1, "\n", s2, "\n", a2, "\n", a3);
}

https://dlang.org/spec/expression.html#SliceExpression

Go

package main
import "fmt"
func main() {
   a1 := []string{"Sun", "Mon", "Tue"}
   // string from front
   s1 := a1[0]
   // string from back
   s2 := a1[len(a1) - 1]
   // array from front
   a2 := a1[:2]
   // array from back
   a3 := a1[len(a1) - 2:]
   // print
   fmt.Println(s1, s2, a2, a3)
}

https://golang.org/ref/spec#Index_expressions
