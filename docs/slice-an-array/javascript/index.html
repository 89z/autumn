Slice an array / JavaScript

Sun Oct 18 2020 array 
let a = ['May', 'June', 'July'];
// example 1
let s1 = a[0];
// example 2
let s2 = a[a.length - 1];
// example 3
let a3 = a.slice(0, 2);
// example 4
let a4 = a.slice(-2);
// print
console.log(s1, s2, a3, a4);
