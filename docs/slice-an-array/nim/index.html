Slice an array / Nim

Wed Aug 5 2020 array 
let a = ["May", "June", "July"]
# example 1
let s1 = a[0]
# example 2
let s2 = a[^1]
# example 3
let a3 = a[0 .. 1]
# example 4
let a4 = a[^2 .. ^1]
# print
echo s1, s2, a3, a4
