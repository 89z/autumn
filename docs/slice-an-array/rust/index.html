fn main() {
   let a = ["M", "a", "r", "c", "h"];
   // example 1
   let s1 = a.get(2);
   // example 2
   let a2 = a.get(2 .. 4);
   // example 3
   let a3 = a.get(2 ..= 3);
   // example 4
   let a4 = a.get(2 ..);
   // print
   println!("{:?} {:?} {:?} {:?}", s1, a2, a3, a4);
}

https://doc.rust-lang.org/std/primitive.slice.html#method.get

fn main() {
   let a = ["M", "a", "r", "c", "h"];

   // example 2
   let a2 = &a[2 .. 4];

   let a3 = &a[2 ..= 3];

}
