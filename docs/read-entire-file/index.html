Go

package main
import (
   "bufio"
   "fmt"
   "os"
)
func main() {
   r1, _ := os.Open("a.txt")
   r2 := bufio.NewScanner(r1)
   var a1 []string
   for r2.Scan() {
      a1 = append(a1, r2.Text())
   }
   fmt.Printf("%q\n", a1)
}

package main
import (
   "fmt"
   "io/ioutil"
)
func main() {
   a1, _ := ioutil.ReadFile("a.txt")
   fmt.Printf("%s", a1)
}

package main
import (
   "io"
   "os"
   "strings"
)
func main() {
   var r1, _ = os.Open("a.txt")
   var s1 strings.Builder
   io.Copy(&s1, r1)
   print(s1.String())
}
