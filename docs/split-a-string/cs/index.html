Split a string / C#

Thu Dec 17 2020 array string 
using C = System.Console;
using J = Newtonsoft.Json.JsonConvert;

class Program {
   static void Main() {
      var s = "May,June,July";
      var a = s.Split(",");
      C.WriteLine(J.SerializeObject(a));
   }
}

https://docs.microsoft.com/dotnet/api/system.string.split
