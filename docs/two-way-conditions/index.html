Two-way conditions

Tue Jun 9 2020 reference Talk:Two-way conditions

D

import std.stdio;
void main() {
   auto n = 0;
   if (n > 0) {
      "positive".writeln;
   } else if (n < 0) {
      "negative".writeln;
   } else {
      "zero".writeln;
   }
}

Dart

main() {

   } else {
      print('zero');
   }
}

Go

package main
func main() {
   n1 := 0
   if n1 > 0 {
      println("positive")
   } else if n1 < 0 {
      println("negative")
   } else {
      println("zero")
   }
}

JavaScript

let n1 = 0;
// example 1
let s1;
if (n1 > 0) {
   s1 = 'positive';
} else if (n1 < 0) {
   s1 = 'negative';
} else {
   s1 = 'zero';
}
// example 2
let s2 = n1 < 0 ? 'negative' : 'positive';
// print
console.log(s1, s2);
