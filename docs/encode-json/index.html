Encode JSON

Dart

import 'dart:convert';
main() {
   var m = {'Sunday': 10};
   var s = jsonEncode(m);
   print(s);
}

https://api.dart.dev/dart-convert/dart-convert-library.html

Go

package main
import (
   "encoding/json"
   "fmt"
)
func main() {
   a := []string{"Sun", "Mon"}
   y, e := json.Marshal(a)
   if e == nil {
      fmt.Printf("%s\n", y)
   }
}

https://golang.org/pkg/encoding/json

JavaScript

let s1, a1 = ['Sun', 'Mon'];
// example 1
s1 = JSON.stringify(a1);
console.log(s1);
// example 2
s1 = JSON.stringify(a1, null, 3);

PHP

<?php
$a1 = ['Sun', 'Mon'];
# example 1
$s1 = json_encode($a1);
# example 2
$s2 = json_encode($a1, JSON_UNESCAPED_SLASHES);
# example 3
$s3 = json_encode($a1, JSON_PRETTY_PRINT | JSON_UNESCAPED_UNICODE);
# print
var_dump($s1, $s2, $s3);
