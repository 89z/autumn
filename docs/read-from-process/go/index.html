package main

import (
   "log"
   "os/exec"
   "strings"
)

func shellExec(name string, arg ...string) (string, error) {
   out, err := exec.Command(name, arg...).Output()
   if err != nil { return "", err }
   return strings.TrimRight(string(out), "\n"), nil
}

func main() {
   out, err := shellExec("go", "version")
   if err != nil {
      log.Fatal(err)
   }
   println(out)
}

package main

import (
   "log"
   "os/exec"
   "strings"
)

func shellExec(name string, arg ...string) (string, error) {
   c, b := exec.Command(name, arg...), new(strings.Builder)
   c.Stdout = b
   e := c.Run()
   return strings.TrimRight(b.String(), "\n"), e
}

func main() {
   s, e := shellExec("go", "version")
   if e != nil {
      log.Fatal(e)
   }
   println(s)
}

package main

import (
   "bufio"
   "log"
   "os/exec"
)

func popen(name string, arg ...string) (*bufio.Scanner, error) {
   cmd := exec.Command(name, arg...)
   out, err := cmd.StdoutPipe()
   if err != nil { return nil, err }
   return bufio.NewScanner(out), cmd.Start()
}

func main() {
   env, err := popen("go", "env")
   if err != nil {
      log.Fatal(err)
   }
   for env.Scan() {
      println(env.Text())
   }
}
