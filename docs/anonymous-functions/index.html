C++

#include <iostream>

int main() {
   // example 1
   auto f1 = [](int n, int n1) -> bool {
      return n > n1;
   };
   // example 2
   auto f2 = [](int n, int n2) {
      return n > n2;
   };
   // print
   auto b1 = f1(9, 8);
   auto b2 = f2(9, 8);
   std::cout << (b1 && b2) << std::endl;
}

D

import std.stdio;

void main() {
   // example 1
   auto f1 = function bool (int n, int n1) {
      return n > n1;
   };
   // example 2
   auto f2 = (int n, int n2) {
      return n > n2;
   };
   // example 3
   auto f3 = function bool (int n, int n3) => n > n3;
   // example 4
   auto f4 = (int n, int n4) => n > n4;
   // print
   auto b1 = f1(9, 8);
   auto b2 = f2(9, 8);
   auto b3 = f3(9, 8);
   auto b4 = f4(9, 8);
   writeln(b1 && b2 && b3 && b4);
}

Rust

fn main() {
   // example 1
   let f1 = |n: u8, n1: u8| -> bool {
      return n > n1;
   };
   // example 2
   let f2 = |n, n2| {
      return n > n2;
   };
   // example 3
   let f3 = |n, n3| n > n3;
   // print
   let b1 = f1(9, 8);
   let b2 = f2(9, 8);
   let b3 = f3(9, 8);
   println!("{}", b1 && b2 && b3);
}
