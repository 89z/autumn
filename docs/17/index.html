Working directory

Wed Jul 8 2020 system

D

import std.file, std.stdio;
void main() {
   auto s = getcwd();
   s.writeln;
}

https://dlang.org/library/std/file/getcwd.html

Dart

import 'dart:io';
main() {
   var o = Directory.current;
   print(o);
}

https://api.dart.dev/dart-io/Directory/current.html

Go

package main
import "os"
func main() {
   s, e := os.Getwd()
   if e != nil {
      os.Exit(1)
   }
   println(s)
}

https://golang.org/pkg/os#Getwd

Python

from os import path
s = path.abspath('.')
print(s)

import os
s = os.getcwd()
print(s)

Rust

use std::{env, io};
fn main() -> io::Result<()> {
   let s = env::current_dir()?;
   println!("{:?}", s);
   Ok(())
}

https://doc.rust-lang.org/std/env/fn.current_dir.html
