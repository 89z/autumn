Object types

Wed Mar 4 18:45:59 2020 object type

Go

package main
import "fmt"
func main() {
   // example 1
   type jan struct {
      sun, mon int
   }
   o1 := jan{10, 11}
   // example 2
   type feb struct {
      sun, mon, tue int
   }
   o2 := feb{mon: 11, tue: 12}
   // example 3
   o3 := struct{sun, mon int}{10, 11}
   // print
   fmt.Println(o1, o2, o3)
}

JavaScript

// example 1
class jan {
   sun = 10;
   mon = 11;
}
let o1 = new jan;
// example 2
let o2 = new class {
   sun = 10;
   mon = 11;
};
// print
console.log(o1, o2);
