Condition-first loops / JavaScript

Fri Sep 11 2020 reference 
console.log('example 1');
for (let n = 10; n < 20; n++) {
   console.log(n);
}

console.log('example 2');
let n = 10;
while (n < 20) {
   console.log(n);
   n++;
}
