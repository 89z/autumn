Condition-first loops

Thu Jun 11 2020 reference

D

import std.stdio;
void main() {
   auto n = 10;
   while (n < 20) {
      n.writeln;
      n++;
   }
}

https://dlang.org/spec/statement.html#while-statement

Dart

main() {
   var n = 10;
   while (n < 20) {
      print(n);
      n++;
   }
}

https://dart.dev/guides/language/language-tour#while-and-do-while

Go

package main
func main() {
   // example 1
   for n1 := 10; n1 < 20; n1++ {
      println(n1)
   }
   // example 2
   n2 := 10
   for n2 < 20 {
      println(n2)
      n2++
   }
}

https://golang.org/ref/spec#For_statements

JavaScript

// example 1
for (let n1 = 10; n1 < 20; n1++) {
   console.log(n1);
}
// example 2
let n2 = 10;
while (n2 < 20) {
   console.log(n2);
   n2++;
}

PHP

<?php
# example 1
for ($n1 = 10; $n1 < 20; $n1++) {
   var_dump($n1);
}
# example 2
$n1 = 10;
while ($n1 < 20) {
   var_dump($n1);
   $n1++;
}
# example 3
$a1 = range(10, 19);
foreach ($a1 as $n1) {
   var_dump($n1);
}
# example 4
function f1($n1) {
   if ($n1 > 19) {
      return;
   }
   var_dump($n1);
   f1($n1 + 1);
}
f1(10);
