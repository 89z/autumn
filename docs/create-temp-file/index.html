Go

package main
import "io/ioutil"
func main() {
   // create only
   f1, _ := ioutil.TempFile("", "")
   s1 := f1.Name()
   // print
   println(s1)
}

PHP

<?php
# create and remove
$r1 = tmpfile();
$s1 = stream_get_meta_data($r1)['uri'];
# create only
$s2 = tempnam('', '');
$r2 = fopen($s2, 'w');
# print
var_dump($s1, $s2);

Python

import tempfile
# create and remove
m1 = tempfile.NamedTemporaryFile()
s1 = m1.name
# create only
m2 = tempfile.NamedTemporaryFile(delete = 0)
s2 = m2.name
# print
print(s1, s2)
