Join an array

D

import std.array, std.stdio;
void main() {
   const a1 = ["Sun", "Mon"];
   const s1 = a1.join(",");
   s1.writeln;
}

Go

package main
import "strings"
func main() {
   a1 := []string{"Sun", "Mon"}
   s1 := strings.Join(a1, ",")
   println(s1)
}
