Join an array / Rust

Mon Sep 14 2020 array string 
fn main() {
   let a = vec!["May", "June"];
   // example 1
   let s1 = a.concat();
   // example 2
   let s2 = a.join(",");
   // print
   println!("{}", s1 == "MayJune" && s2 == "May,June");
}
