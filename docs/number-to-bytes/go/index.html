Number to bytes / Go

Fri Jul 2 2021 encoding 
package main

import (
   "encoding/binary"
   "fmt"
   "math/bits"
)

func encodeUint(x uint64) []byte {
   buf := make([]byte, 8)
   binary.BigEndian.PutUint64(buf, x)
   return buf[bits.LeadingZeros64(x) >> 3:]
}

func main() {
   buf := encodeUint(0xFF)
   fmt.Println(buf) // [255]
}

package main

import (
   "bytes"
   "encoding/binary"
   "fmt"
)

func main() {
   var (
      b = new(bytes.Buffer)
      n uint16 = 0xFF
   )
   binary.Write(b, binary.BigEndian, n)
   fmt.Println(b.Bytes()) // [0 255]
}
