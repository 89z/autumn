Get map value

Fri May 1 22:48:28 2020 map

Go

package main
func main() {
   m1 := map[string]int{"Sun": 10, "Mon": 11}
   // example 1
   n1, _ := m1["Mon"]
   // example 2
   n2 := m1["Mon"]
   // print
   println(n1, n2)
}

https://golang.org/ref/spec#Index_expressions

JavaScript

let m1 = {Sun: 10, Mon: 11};
// example 1
let n1 = m1['Mon'];
// example 2
let n2 = m1.Mon;
// print
console.log(n1, n2);
https://developer.mozilla.org/Web/JavaScript/Reference/Operators/Property_Accessors
