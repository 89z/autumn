Get map value

Wed Mar 4 18:45:59 2020 map

Go

package main
func main() {
   m1 := map[string]int{"sun": 10, "mon": 11}
   // example 1
   n1, _ := m1["mon"]
   // example 2
   n2 := m1["mon"]
   // print
   println(n1, n2)
}

https://golang.org/ref/spec#Index_expressions

JavaScript

let m1 = {sun: 10, mon: 11};
// example 1
let n1 = m1['mon'];
// example 2
let n2 = m1.mon;
// print
console.log(n1, n2);
https://developer.mozilla.org/Web/JavaScript/Reference/Operators/Property_Accessors

Julia

m1 = Dict("sun" => 10, "mon" => 11)
n1 = get(m1, "sun", 0)
println(n1)
