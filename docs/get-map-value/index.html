Dart

main() {
   var m = {'Sunday': 10};
   var n = m['Sunday'];
   print(n);
}

Go

package main
func main() {
   m1 := map[string]int{"Sunday": 10}
   // example 1
   n1, b1 := m1["Sunday"]
   // example 2
   n2 := m1["Sunday"]
   // print
   println(n1, b1, n2)
}

https://golang.org/ref/spec#Index_expressions

JavaScript

let m1 = {Sun: 10, Mon: 11};
// example 1
let n1 = m1['Mon'];
// example 2
let n2 = m1.Mon;
// print
console.log(n1, n2);
