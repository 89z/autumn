Get map value

Fri Sep 11 2020 map

D

import std.stdio;

void main() {
   auto m = ["year": 2019];
   auto n = m["year"];
   writeln(n == 2019);
}

https://dlang.org/spec/hash-map.html

Dart

void main() {
   var m = {'year': 2019};
   var n = m['year'];
   print(n == 2019);
}

https://api.dart.dev/dart-core/Map/operator_get.html

Go

package main

func main() {
   m := map[string]int{"year": 2019}
   // example 1
   n, b := m["year"]
   // example 2
   n2 := m["year"]
   // print
   println(b, n == 2019, n2 == 2019)
}

https://golang.org/ref/spec#Index_expressions

JavaScript

let m = {year: 2019};
// example 1
let n = m['year'];
// example 2
let n2 = m.year;
// print
console.log(n == 2019, n2 == 2019);

Rust

use std::collections::HashMap;

fn main() {
   let mut m = HashMap::new();
   m.insert("year", 2019);
   let n = m["year"];
   println!("{}", n == 2019);
}
