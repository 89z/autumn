Iterate a process

Fri Sep 11 2020 iterate input-output

Go

package main
import (
   "bytes"
   "fmt"
   "os"
   "os/exec"
)
func f(r rune) bool {
   return r == '\n'
}
func main() {
   y, e := exec.Command("go", "env").Output()
   if e != nil {
      os.Exit(1)
   }
   a := bytes.FieldsFunc(y, f)
   for n, y := range a {
      fmt.Printf("%v %s\n", n, y)
   }
}

package main

import (
   "bufio"
   "bytes"
   "os"
   "os/exec"
)

func main() {
   y, e := exec.Command("go", "env").Output()
   if e != nil {
      os.Exit(1)
   }
   o := bytes.NewReader(y)
   o2 := bufio.NewScanner(o)
   for o2.Scan() {
      s := o2.Text()
      println(s)
   }
}

PHP

<?php
$s = 'php -v';
# example 1
$r = popen($s, 'r');
while (true) {
   $s2 = fgets($r);
   if (feof($r)) {
      break;
   }
   var_dump($s2);
}
# example 2
exec($s, $a);
foreach ($a as $s2) {
   var_dump($s2);
}

Python

import subprocess
a = ['ag', '-V']
o = subprocess.Popen(a, stdout=subprocess.PIPE, text=True)
for s in o.stdout:
   print(s, end='')
