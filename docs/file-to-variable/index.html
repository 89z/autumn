Go

package main

import (
   "io"
   "os"
   "strings"
)

func main() {
   o, e := os.Open("a.txt")
   if e != nil {
      os.Exit(1)
   }
   o1 := strings.Builder{}
   io.Copy(&o1, o)
   s := o1.String()
   print(s)
}

package main

import (
   "fmt"
   "io/ioutil"
   "log"
   "os"
)

func main() {
   o, e := os.Open("a.txt")
   if e != nil {
      log.Fatal(e)
   }
   y, e := ioutil.ReadAll(o)
   if e != nil {
      log.Fatal(e)
   }
   fmt.Printf("%s", y)
}

package main

import (
   "bufio"
   "fmt"
   "os"
)

func main() {
   o_open, e := os.Open("a.txt")
   if e != nil {
      os.Exit(1)
   }
   o_scan := bufio.NewScanner(o_open)
   a := []string{}
   for o_scan.Scan() {
      s := o_scan.Text()
      a = append(a, s)
   }
   fmt.Println(a)
}
