using System;

class Program {
   static void Main() {
      // example 1
      int n1 = 1_000;
      // example 2
      uint n2 = 10;
      // example 3
      long n3 = 10;
      // example 4
      ulong n4 = 10;
      // print
      Console.WriteLine("{0} {1} {2} {3}", n1, n2, n3, n4);
   }
}
