PHP

<?php
function powerset($a_in) {
   $a_out = [[]];
   foreach ($a_in as $a_elem) {
      foreach ($a_out as $a_comb) {
         $a_out[] = array_merge([$a_elem], $a_comb);
      }
   }
   return $a_out;
}
$a1 = ['Sun', 'Mon', 'Tue'];
$a2 = powerset($a1);
var_export($a2);

def powerset(a_in):
   a_out = [[]]
   for v_in in a_in:
      a_out += [a_each + [v_in] for a_each in a_out]
   return a_out
a1 = ['Sun', 'Mon', 'Tue']
