C#

using System;
using System.IO;

class Program {
   static void Main() {
      var s = "Program.cs";
      var s1 = Path.GetFullPath(s);
      Console.WriteLine(s1);
   }
}

   auto s = "index.md";

   s1.writeln;
}

Go

package main

import (
   "os"
   "path/filepath"
)

func main() {
   s := "index.md"
   s1, e := filepath.Abs(s)
   if e != nil {
      os.Exit(1)
   }
   println(s1)
}

https://golang.org/pkg/path/filepath#Abs
