C++

#include <unistd.h>

   int n = access("index.md", F_OK);
   std::cout << (n != -1) << std::endl;

Dart

import 'dart:io';

void main() {
   var o = new File('index.md');
   var b = o.existsSync();
   print(b);
}

Go

package main
import "os"

func main() {
   o, e := os.Stat("index.md")
   // example 1
   if e != nil {
      os.Exit(1)
   }
   // example 2
   b := o.IsDir()
   println(b)
}

package main
import "os"

func main() {
   o, e := os.Stat("index.md")
   // example 1
   if os.IsNotExist(e) {
      os.Exit(1)
   }
   // example 2
   b2 := o.Mode().IsRegular()
   // example 3
   b3 := ! o.Mode().IsDir()
   // print
   println(b2, b3)
}

  • https://golang.org/pkg/os#IsNotExist
  • https://golang.org/pkg/os#Stat
