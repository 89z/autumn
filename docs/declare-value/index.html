Go

package main
func main() {
   // example 1
   const n1 int = 10
   // example 2
   const n2 = 10
   // example 3
   var n3 int = 10
   // example 4
   var n4 = 10
   // example 5
   var n5 int
   // example 6
   n6 := 10
   // print
   println(n1, n2, n3, n4, n5, n6)
}
