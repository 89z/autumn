Filename to extension

Mon Aug 31 2020 string

C++

#include <filesystem>
#include <iostream>
int main() {
   auto s1 = "C:\\Windows\\write.exe";
   auto o1 = std::filesystem::path(s1);
   auto s2 = o1.extension();
   std::cout << s2 << std::endl;
}

https://en.cppreference.com/w/cpp/filesystem/path/extension

D

import std.path, std.stdio;
void main() {
   auto s1 = "index.md";
   auto s2 = s1.extension;
   writeln(s2 == ".md");
}

https://dlang.org/library/std/path/extension.html

Go

package main
import "path"
func main() {
   s1 := "a.go"
   s2 := path.Ext(s1)
   println(s2 == ".go")
}

package main
import "path/filepath"
func main() {
   s1 := "a.go"
   s2 := filepath.Ext(s1)
   println(s2 == ".go")
}
