Filename to extension

Sun Aug 30 2020 string

Go

package main
import "path"
func main() {
   s1 := "a.go"
   s2 := path.Ext(s1)
   println(s2 == ".go")
}

package main
import "path/filepath"
func main() {
   s1 := "a.go"
   s2 := filepath.Ext(s1)
   println(s2 == ".go")
}

PHP

<?php
$s = 'a.php';
# example 1
$s1 = pathinfo($s, PATHINFO_EXTENSION);
# example 2
$o = new SplFileInfo($s);
$s2 = $o->getExtension();
# print
var_dump($s1 == 'php', $s2 == 'php');
