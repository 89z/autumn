HTTP to variable / JavaScript

Fri Sep 11 2020 input-output network 

let s_uri = 'https://speedtest.lax.hivelocity.net';

fetch(s_uri).then(
   o_req => o_req.text()
).then(
   s_resp => console.log(s_resp)
);

async function main() {
   let s_uri = 'https://speedtest.lax.hivelocity.net';
   let o_req = await fetch(s_uri);
   let s_resp = await o_req.text();
   console.log(s_resp);
}

main();
