HTTP to variable / JavaScript

Mon Nov 9 2020 input-output 
let uri_s = 'http://speedtest.lax.hivelocity.net';

fetch(uri_s).then(
   req_o => req_o.text()
).then(
   resp_s => console.log(resp_s)
);

https://developer.mozilla.org/Web/API/WindowOrWorkerGlobalScope/fetch

async function main() {
   let uri_s = 'https://speedtest.lax.hivelocity.net';
   let req_o = await fetch(uri_s);
   let resp_s = await req_o.text();
   console.log(resp_s);
}

main();
