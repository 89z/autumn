HTTP to variable / Go

Mon Nov 9 2020 input-output network 
package main

import (
   "bytes"
   "log"
   "net/http"
)

func main() {
   o_in, e := http.Get("https://speedtest.lax.hivelocity.net")
   if e != nil {
      log.Fatal(e)
   }
   o_out := bytes.Buffer{}
   n, e := o_out.ReadFrom(o_in.Body)
   if e != nil {
      log.Fatal(e)
   }
   print(n, o_out.String())
}

package main

import (
   "io"
   "os"
   "strings"
)

func main() {
   o, e := os.Open("a.txt")
   if e != nil {
      os.Exit(1)
   }
   o1 := strings.Builder{}
   io.Copy(&o1, o)
   s := o1.String()
   print(s)
}

https://golang.org/pkg/io#Copy

package main

import (
   "fmt"
   "io/ioutil"
   "log"
   "net/http"
)

func main() {
   o, e := http.Get("https://speedtest.lax.hivelocity.net")
   if e != nil {
      log.Fatal(e)
   }
   y, e := ioutil.ReadAll(o.Body)
   if e != nil {
      log.Fatal(e)
   }
   fmt.Printf("%s", y)
}

https://golang.org/pkg/io/ioutil#ReadAll
