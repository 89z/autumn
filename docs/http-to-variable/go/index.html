HTTP to variable / Go

Sat Mar 6 2021 network 
package main

import (
   "log"
   "net/http"
   "net/url"
   "os"
)

func main() {
   req := &http.Request{
      URL: &url.URL{Scheme: "http", Host: "speedtest.lax.hivelocity.net"},
   }
   resp, e := http.DefaultClient.Do(req)
   if e != nil {
      log.Fatal(e)
   }
   os.Stdout.ReadFrom(resp.Body)
}

https://golang.org/pkg/net/http#Client.Do

package main

import (
   "io/ioutil"
   "log"
   "net/http"
   "os"
)

func main() {
   get, e := http.Get("http://speedtest.lax.hivelocity.net")
   if e != nil {
      log.Fatal(e)
   }
   b, e := ioutil.ReadAll(get.Body)
   if e != nil {
      log.Fatal(e)
   }
   os.Stdout.Write(b)
}
