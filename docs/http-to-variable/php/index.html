HTTP to variable / PHP

Fri Sep 11 2020 input-output network 

<?php
$s = 'https://speedtest.lax.hivelocity.net';
# example 1
$s1 = file_get_contents($s);
# example 2
$r = fopen($s, 'r');
$s2 = stream_get_contents($r);
# print
var_dump($s1, $s2);

<?php
extension_loaded('curl') or die('curl');
extension_loaded('openssl') or die('openssl');
$r = curl_init('https://speedtest.lax.hivelocity.net');
curl_setopt($r, CURLOPT_RETURNTRANSFER, true);
$s = curl_exec($r);
echo $s;
