Decode Gzip

Sat May 2 19:33:52 2020 file-format

Go

package main
import (
   "compress/gzip"
   "io"
   "os"
)
func main() {
   o1, _ := os.Open("a.tar.gz")
   o2, _ := gzip.NewReader(o1)
   o3, _ := os.Create("a.tar")
   // begin
   io.Copy(o3, o2)
}

https://golang.org/pkg/compress/gzip

PHP

<?php
extension_loaded('zlib') or die('php-zlib');
$s1 = file_get_contents('a.tar.gz');
$s2 = gzdecode($s1);
file_put_contents('a.tar', $s2);
