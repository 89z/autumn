Iterate a process / Go

Fri Sep 11 2020 iterate input-output 
package main

import (
   "bufio"
   "bytes"
   "os"
   "os/exec"
)

func main() {
   y, e := exec.Command("go", "env").Output()
   if e != nil {
      os.Exit(1)
   }
   o := bytes.NewReader(y)
   o1 := bufio.NewScanner(o)
   for o1.Scan() {
      s := o1.Text()
      println(s)
   }
}

package main

import (
   "bytes"
   "fmt"
   "os"
   "os/exec"
)

func f(r rune) bool {
   return r == '\n'
}

func main() {
   y, e := exec.Command("go", "env").Output()
   if e != nil {
      os.Exit(1)
   }
   a := bytes.FieldsFunc(y, f)
   for n, y := range a {
      fmt.Printf("%v %s\n", n, y)
   }
}
