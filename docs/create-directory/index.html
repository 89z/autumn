Create directory

Tue Jul 14 2020 system

D

import std.file;
void main() {
   "Sunday".mkdir;
}

https://dlang.org/library/std/file/mkdir.html

Dart

import 'dart:io';
main() {
   // example 1
   var o1 = new Directory('Sunday');
   o1.createSync();
   // example 2
   var o2 = new Directory('Monday/Tuesday');
   o2.createSync(recursive: true);
}

https://api.dart.dev/dart-io/Directory-class.html

Go

package main
import "os"
func main() {
   // example 1
   os.Mkdir("Sunday", os.ModeDir)
   // exmaple 2
   os.MkdirAll("Monday/Tuesday", os.ModeDir)
}

Rust

use std::fs;
fn main() {
   // example 1
   let e1 = fs::create_dir("Sunday");
   println!("{:?}", e1);
   // example 2
   let e2 = fs::create_dir_all("Sunday/Monday");
   println!("{:?}", e2);
}
