package main

import (
   "fmt"
   "math"
)

func numberFormat(d float64) string {
   e := int(math.Log10(d)) / 3
   f := d / math.Pow10(e * 3)
   return fmt.Sprintf("%.3f", f) + []string{1: " k", " M", " G"}[e]
}

func main() {
   s := numberFormat(9012345678)
   println(s == "9.012 G")
}

package main
import "fmt"

func numberFormat(d float64) string {
   var e int
   for d >= 1000 {
      d /= 1000
      e++
   }
   return fmt.Sprintf("%.3f", d) + []string{1: " k", " M", " G"}[e]
}

func main() {
   s := numberFormat(9012345678)
   println(s == "9.012 G")
}
