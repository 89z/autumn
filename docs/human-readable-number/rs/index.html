fn number_format(n: f64) -> String {
   let (mut n2, mut n3) = (n, 0);
   while n2 >= 1e3 {
      n2 /= 1e3;
      n3 += 1;
   }
   format!("{:.3}", n2) + ["", " k", " M", " G"][n3]
}

fn main() {
   let s = number_format(9012345678e0);
   println!("{}", s == "9.012 G");
}

fn number_format(n: f64) -> String {
   let n2 = n.log(1e3);
   let n3 = n / 1e3f64.powi(n2 as i32);
   format!("{:.3}", n3) + ["", " k", " M", " G"][n2 as usize]
}

fn main() {
   let s = number_format(9012345678e0);
   println!("{}", s == "9.012 G");
}
