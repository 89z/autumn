using System;

class Program {
   static void Main() {
      string[] suf = { " ", "K", "M", "G" };
      long bytes = 1264;
      int place = Convert.ToInt32(Math.Floor(Math.Log(bytes, 1024)));
      double num = Math.Round(bytes / Math.Pow(1024, place), 3);
      var s = num.ToString() + ' ' + suf[place];
      Console.WriteLine(s == "1.234 K");
   }
}

using System;

class Program {
   static string HumanReadable(double len) {
      int order = 0;
      while (len >= 1024) {
         order++;
         len /= 1024;
      }
      return String.Format("{0:0.###}", len) + ' ' + " KMG"[order];
   }

   static void Main() {
      var s = HumanReadable(1264);
      Console.WriteLine(s == "1.234 K");
   }
}
