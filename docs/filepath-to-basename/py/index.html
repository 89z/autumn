from os import path
s = r'C:\Windows\notepad.exe'
s2 = path.basename(s)
print(s2 == 'notepad.exe')

import pathlib
o = pathlib.Path(r'C:\Windows\notepad.exe')
s = o.name
print(s == 'notepad.exe')
