Filepath to basename / Go

Thu Sep 10 2020 system 

package main
import "path"

func main() {
   s := "C:/go/bin/go.exe"
   s1 := path.Base(s)
   println(s1 == "go.exe")
}

package main
import "path/filepath"

func main() {
   s := `C:\Windows\notepad.exe`
   s1 := filepath.Base(s)
   println(s1 == "notepad.exe")
}
