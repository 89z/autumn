Filepath to basename

Sun Aug 30 2020 string

Go

package main
import "path"
func main() {
   s1 := "C:/go/bin/go.exe"
   s2 := path.Base(s1)
   println(s2 == "go.exe")
}

package main
import "path/filepath"
func main() {
   s1 := `C:\go\bin\go.exe`
   s2 := filepath.Base(s1)
   println(s2 == "go.exe")
}

PHP

<?php
$o = new SplFileInfo('C:\\php\\license.txt');
# example 1
$s1 = $o->getBasename();
# example 2
$s2 = $o->getFilename();
# example 3
$s3 = $o->getBasename('.txt');
# print
var_dump($s1 == 'license.txt', $s2 == 'license.txt', $s3 == 'license');

<?php
$s = 'C:\\php\\license.txt';
# example 1
$s1 = basename($s);
# example 2
$s2 = pathinfo($s, PATHINFO_BASENAME);
# example 3
$s3 = basename($s, '.txt');
# example 4
$s4 = pathinfo($s, PATHINFO_FILENAME);
# print
var_dump(
   $s1 == 'license.txt',
   $s2 == 'license.txt',
   $s3 == 'license',
   $s4 == 'license'
);

  • https://php.net/function.basename
  • https://php.net/function.pathinfo

Python

from os import path
s1 = 'C:\\python\\python.exe'
s2 = path.basename(s1)
print(s2 == 'python.exe')
