C#

using System;
using System.IO;

class Program {
   static void Main() {
      var s = @"C:\Windows\notepad.exe";
      var s1 = Path.GetFileName(s);
      Console.WriteLine(s1 == "notepad.exe");
   }
}

Go

package main
import "path"

func main() {
   s := "C:/go/bin/go.exe"
   s1 := path.Base(s)
   println(s1 == "go.exe")
}

package main
import "path/filepath"

func main() {
   s := `C:\Windows\notepad.exe`
   s1 := filepath.Base(s)
   println(s1 == "notepad.exe")
}

PHP

<?php
$o = new SplFileInfo('C:\\Windows\\notepad.exe');
# example 1
$s1 = $o->getBasename();
# example 2
$s2 = $o->getFilename();
# example 3
$s3 = $o->getBasename('.exe');
# print
var_dump($s1 == 'notepad.exe', $s2 == 'notepad.exe', $s3 == 'notepad');

<?php
$s = 'C:\\Windows\\notepad.exe';
# example 1
$s1 = basename($s);
# example 2
$s2 = pathinfo($s, PATHINFO_BASENAME);
# example 3
$s3 = basename($s, '.exe');
# example 4
$s4 = pathinfo($s, PATHINFO_FILENAME);
# print
var_dump(
   $s1 == 'notepad.exe',
   $s2 == 'notepad.exe',
   $s3 == 'notepad',
   $s4 == 'notepad'
);

  • https://php.net/function.basename
