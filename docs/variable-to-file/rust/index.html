Variable to file / Rust

Fri Sep 11 2020 input-output 

use std::{io, fs};

fn main() -> io::Result<()> {
   let s = "May\n";
   fs::write("a.txt", s)?;
   Ok(())
}

use std::fs::File;
use std::io::{Error, Write};

fn main() -> Result<(), Error> {
   let y = b"May\n";
   let mut o = File::create("a.txt")?;
   o.write(y)?;
   Ok(())
}
