Fri Sep 11 2020 input-output 
package main

import (
   "fmt"
   "os"
)

func main() {
   s := "March"
   o, e := os.Create("a.txt")
   if e != nil {
      os.Exit(1)
   }
   fmt.Fprintln(o, s)
}

https://golang.org/pkg/fmt#Fprintln

package main
import "os"

func main() {
   s := "March\n"
   o, e := os.Create("a.txt")
   if e != nil {
      os.Exit(1)
   }
   o.WriteString(s)
}

https://golang.org/pkg/os#File.WriteString

package main

import (
   "os"
   "strings"
)

func main() {
   r_o := strings.NewReader("March\n")
   w_o, e := os.Create("a.txt")
   if e != nil {
      os.Exit(1)
   }
   r_o.WriteTo(w_o)
}

https://golang.org/pkg/strings#Reader.WriteTo
