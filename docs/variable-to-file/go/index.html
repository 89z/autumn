Variable to file / Go

Fri Sep 11 2020 input-output 

package main

import (
   "fmt"
   "os"
)

func main() {
   s := "May"
   o, e := os.Create("a.txt")
   if e != nil {
      os.Exit(1)
   }
   fmt.Fprintln(o, s)
}

package main
import "os"

func main() {
   s := "May\n"
   o, e := os.Create("a.txt")
   if e != nil {
      os.Exit(1)
   }
   o.WriteString(s)
}

package main

import (
   "os"
   "strings"
)

func main() {
   o := strings.NewReader("May\n")
   o1, e := os.Create("a.txt")
   if e != nil {
      os.Exit(1)
   }
   o.WriteTo(o1)
}
