Go

package main

import (
   "fmt"
   "os"
)

func main() {
   s := "May"
   o, e := os.Create("a.txt")
   if e != nil {
      os.Exit(1)
   }
   fmt.Fprintln(o, s)
}

package main
import "os"

func main() {
   s := "May\n"
   o, e := os.Create("a.txt")
   if e != nil {
      os.Exit(1)
   }
   o.WriteString(s)
}

package main

import (
   "os"
   "strings"
)

func main() {
   o := strings.NewReader("May\n")
   o1, e := os.Create("a.txt")
   if e != nil {
      os.Exit(1)
   }
   o.WriteTo(o1)
}

Rust

use std::{io, fs};

fn main() -> io::Result<()> {
   let s = "May\n";
   fs::write("a.txt", s)?;
   Ok(())
}

use std::fs::File;
use std::io::{Error, Write};

fn main() -> Result<(), Error> {
   let y = b"May\n";
   let mut o = File::create("a.txt")?;
   o.write(y)?;
   Ok(())
}
