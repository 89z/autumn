Variadic function

Wed Mar 25 10:17:37 2020 function

Go

package main
import "strings"
func f1(a1... string) string {
   return strings.Join(a1, ",")
}
func main() {
   s1 := f1("Sun", "Mon")
   println(s1)
}

https://golang.org/ref/spec#Function_types

JavaScript

function f1(...a1) {
   return a1.join(' ');
}
console.log(f1);

PHP

<?php
# example 1
function f1() {
   $a1 = func_get_args();
   return implode(',', $a1);
}
$s1 = f1('Sun', 'Mon');
# example 2
function f2(...$a1) {
   return implode(',', $a1);
}
$s2 = f2('Sun', 'Mon');
# print
var_dump($s1, $s2);
  • https://php.net/function.func-get-args
  • https://php.net/functions.arguments

Python

def f1(*a1):
   return ','.join(a1)
s1 = f1('Sun', 'Mon')
print(s1)
