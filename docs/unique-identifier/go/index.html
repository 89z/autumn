package main
import "math/big"

func Decode(s string, n int) int64 {
   o := big.Int{}
   o.SetString(s, n)
   return o.Int64()
}

func main() {
   s := "qm8rbz"
   n := Decode(s, 36)
   println(n == 1609480799)
}

package main
import "math/big"

func Encode36(x int) string {
   n := int64(x)
   return big.NewInt(n).Text(36)
}

func main() {
   n := 1609480799
   s := Encode36(n)
   println(s == "qm8rbz")
}

package main
import "strconv"

func Encode36(x int) string {
   n := int64(x)
   return strconv.FormatInt(n, 36)
}

func main() {
   n := 1609480799
   s := Encode36(n)
   println(s == "qm8rbz")
}
