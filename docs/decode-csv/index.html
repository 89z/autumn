Go

package main
import (
   "encoding/csv"
   "os"
)
func main() {
   r1, _ := os.Open("a.csv")
   r2 := csv.NewReader(r1)
   m1 := map[string]int{}
   for {
      a1, e1 := r2.Read()
      if e1 != nil {
         break
      }
      if len(m1) == 0 {
         for n1, s1 := range a1 {
            m1[s1] = n1
         }
         continue
      }
      println(a1[m1["city"]])
   }
}

PHP

<?php
$r1 = fopen('a.csv', 'r');
$a1 = [];
while (true) {
   $a2 = fgetcsv($r1);
   if (feof($r1)) {
      break;
   }
   if ($a1 == []) {
      $a1 = $a2;
      continue;
   }
   $m1 = array_combine($a1, $a2);
   echo $m1['city'], "\n";
}
