Go

package main

import (
   "encoding/csv"
   "fmt"
   "log"
   "os"
)

func main() {
   o_csv, e := os.Open("a.csv")
   if e != nil {
      log.Fatal(e)
   }
   o_table := csv.NewReader(o_csv)
   a_head, e := o_table.Read()
   if e != nil {
      log.Fatal(e)
   }
   a_body, e := o_table.ReadAll()
   if e != nil {
      log.Fatal(e)
   }
   m_row := map[string]string{}
   for n_row, a_row := range a_body {
      for n_col, s_col := range a_head {
         m_row[s_col] = a_row[n_col]
      }
      fmt.Println(n_row, m_row)
   }
}

<?php
$r_tab = fopen('a.csv', 'r');
$a_head = fgetcsv($r_tab);

while (true) {
   $a_row = fgetcsv($r_tab);
   if (feof($r_tab)) {
      break;
   }
   $m_row = array_combine($a_head, $a_row);
   $s_city = $m_row['city'];
   var_dump($s_city);
}

import csv
o_db = open('a.csv')
o_tab = csv.reader(o_db)
a_head = next(o_tab)

for a_row in o_tab:
   m_row = dict(zip(a_head, a_row))
   s_city = m_row['city']
   print(s_city)
