package main
import "fmt"

func main() {
   n := 100
   { // example 1
      s := fmt.Sprintf("%d", n)
      fmt.Println(s == "100")
   }
   { // example 2
      s := fmt.Sprintf("%v", n)
      fmt.Println(s == "100")
   }
}

package main
import "strconv"

func main() {
   b := strconv.AppendInt(nil, 100, 10)
   println(string(b) == "100")
}

package main
import "strconv"

func main() {
   var n int64 = 100
   s := strconv.FormatInt(n, 10)
   println(s == "100")
}
