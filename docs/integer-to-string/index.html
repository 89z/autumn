Go

package main
import "fmt"

func main() {
   n := 9
   // example 1
   s1 := fmt.Sprint(n)
   // example 2
   s2 := fmt.Sprintf("%d", n)
   // print
   println(s1 == "9", s2 == "9")
}

package main
import "strconv"

func main() {
   // example 1
   n := int64(9)
   s1 := strconv.FormatInt(n, 10)
   // example 2
   n2 := 9.9
   s2 := strconv.FormatFloat(n2, 'f', -1, 64)
   // print
   println(s1 == "9", s2 == "9.9")
}
