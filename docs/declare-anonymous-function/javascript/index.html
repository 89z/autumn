Declare anonymous function / JavaScript

Tue Sep 15 2020 function 
// example 1
let f1 = function (n, n1) {
   return n > n1;
};
// example 2
let f2 = (n, n2) => n > n2;
// print
let b1 = f1(9, 8);
let b2 = f2(9, 8);
console.log(b1, b2);
