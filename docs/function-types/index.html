Function types

Thu May 14 2020 function

Go

package main
import "fmt"
// example 1
func f1(n1 int) int {
   return n1 + 10
}
// example 2
var f2 = func(n1 int) int {
   return n1 + 10
}
// print
func main() {
   fmt.Printf("%T\n", f1)
   fmt.Printf("%T\n", f2)
}

https://golang.org/ref/spec#Function_types

JavaScript

// example 1
function f1(n1) {
   return n1 + 10;
}
// example 2
let f2 = function (n1) {
   return n1 + 10;
};
// example 3
let f3 = n1 => n1 + 10;
// print
console.log(f1, f2, f3);

PHP

<?php
# example 1
function f1($s1) {
   return intval($s1);
}
$n1 = f1('10');
# example 2
$f2 = function ($s1) {
   return intval($s1);
};
$n2 = $f2('10');
# example 3
$f3 = fn($s1) => intval($s1);
$n3 = $f3('10');
# print
var_dump($n1, $n2, $n3);
