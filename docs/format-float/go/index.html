Format float / Go

Fri Sep 11 2020 number 

package main
import "fmt"

func main() {
   n := 9.9
   // example 1
   fmt.Printf("%.0f\n", n)
   // example 2
   s := fmt.Sprintf("%.0f", n)
   println(s)
}

package main

import (
   "fmt"
   "os"
)

func main() {
   n := 9.9
   fmt.Fprintf(os.Stdout, "%.0f\n", n)
}

package main
import "strconv"

func main() {
   n := 9.9
   // example 1
   s1 := strconv.FormatFloat(n, 'f', -1, 64)
   // example 2
   s2 := strconv.FormatFloat(n, 'f', 0, 64)
   // print
   println(s1 == "9.9", s2 == "10")
}
