C++

int main() {
   auto n = 9.9;
   std::printf("%.0f\n", n);
}

D

import std.stdio;
 
void main() {
    immutable r = 7.125;
    writefln(" %9.3f",  -r);
    writefln(" %9.3f",   r);
    writefln(" %-9.3f",  r);
    writefln(" %09.3f", -r);
    writefln(" %09.3f",  r);
    writefln(" %-09.3f", r);
}

https://dlang.org/library/std/stdio/writefln.html
