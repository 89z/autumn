Match string with pattern

Tue May 26 2020 string

Go

package main
import "regexp"
func main() {
   s1 := "Sunday"
   // example 1
   b1 := regexp.MustCompile("^Su").MatchString(s1)
   // example 2
   b2 := regexp.MustCompile("un.").MatchString(s1)
   // example 3
   b3 := regexp.MustCompile("ay$").MatchString(s1)
   // example 4
   b4 := regexp.MustCompile("(?i)su").MatchString(s1)
   // print
   println(b1, b2, b3, b4)
}

PHP

<?php
$s1 = 'Sunday';
# example 1
$n1 = preg_match('/^Su/', $s1);
# example 2
$n2 = preg_match('/un./', $s1);
# example 3
$n3 = preg_match('/ay$/', $s1);
# example 4
$n4 = preg_match('/su/i', $s1);
# example 5
$n5 = preg_match('/(?i)su/', $s1);

var_dump($n1 == 1, $n2 == 1, $n3 == 1, $n4 == 1, $n5 == 1);

Python

import re
s1 = 'Sunday'
# example 1
o1 = re.search('^Su', s1)
# example 2
o2 = re.search('un.', s1)
# example 3
o3 = re.search('ay$', s1)
# example 4
o4 = re.search('su', s1, re.I)
# example 5
o5 = re.search('(?i)su', s1)
# print
print(o1 != None, o2 != None, o3 != None, o4 != None, o5 != None)
  • https://docs.python.org/library/re.html#re.I
  • https://docs.python.org/library/re.html#re.search
