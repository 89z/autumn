Go

package main
import (
   "fmt"
   "log"
   "time"
)
func main() {
   o1, e := time.Parse(time.RFC3339[:10], "2019-12-31")
   if e != nil {
      log.Fatal(e)
   }
   o2 := time.Now()
   o3 := o2.Sub(o1)
   n := o3.Seconds()
   fmt.Println(n)
}

Python

from datetime import datetime
o1 = datetime.strptime('2019-12-31', '%Y-%m-%d')
o2 = datetime.now()
o3 = o2 - o1
n = o3.total_seconds()
print(n)
