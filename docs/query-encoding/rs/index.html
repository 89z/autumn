use form_urlencoded::Serializer;

fn encode(m: &[(&str, &str)]) -> String {
   let s = String::new();
   Serializer::new(s).extend_pairs(m).finish()
}

fn main() {
   let m = &[("month", "May"), ("day", "Friday")];
   let s = encode(m);
   println!("{}", s);
}

use std::collections::HashMap;

fn decode(s: &str) -> HashMap<String, String> {
   let a = s.as_bytes();
   form_urlencoded::parse(a).into_owned().collect()
}

fn main() {
   let s = "month=May&day=Friday";
   let m = decode(s);
   println!("{:?}", m);
}
