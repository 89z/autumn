Get directory name

Fri May 1 22:48:28 2020 string

Go

package main
import "path"
func main() {
   s1 := "/sunday/monday.tar.xz"
   // example 1
   b1 := path.Dir(s1) == "/sunday"
   // example 2
   s2, _ := path.Split(s1)
   b2 := s2 == "/sunday/"
   // print
   println(b1, b2)
}
  • https://golang.org/pkg/path#Dir
  • https://golang.org/pkg/path#Split
package main
import "path/filepath"
func main() {
   s1 := `C:\sunday\monday.tar.xz`
   // example 1
   b1 := filepath.Dir(s1) == `C:\sunday`
   // example 2
   s2, _ := filepath.Split(s1)
   b2 := s2 == `C:\sunday\`
   // print
   println(b1, b2)
}

PHP

<?php
$s1 = '/sunday/monday.tar.xz';
# example 1
$b1 = dirname($s1) == '/sunday';
# example 2
$b2 = pathinfo($s1, PATHINFO_DIRNAME) == '/sunday';
# print
var_dump($b1, $b2);
  • https://php.net/function.dirname
  • https://php.net/function.pathinfo

Python

from os import path
s1 = '/sunday/monday.tar.xz'
s2 = path.dirname(s1)
