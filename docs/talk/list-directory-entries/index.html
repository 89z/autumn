fn main() {
   // example 1
   let s1 = "Sunday";
   println!("{}", s1);
   // example 2
   let s2 = String::from("Sunday");
   println!("{}", s2);
}

use std::ffi::OsStr;
fn main() {
   let s1 = OsStr::new("Sunday");
   // std::ffi::OsStr doesn't implement std::fmt::Display
   println!("{}", s1);
}

use std::ffi::OsString;
fn main() {
   let s1 = OsString::from("Sunday");
   // std::ffi::OsString doesn't implement std::fmt::Display
   println!("{}", s1);
}

use std::fs;
fn main() {
   match fs::read_dir(".") {
      Err(e) => println!("{}", e),
      Ok(readdir) => for r_dirent in readdir {
         match r_dirent {
            Err(e) => println!("{}", e),
            // std::fs::DirEntry doesn't implement std::fmt::Display
            Ok(dirent) => println!("{}", dirent)
         }
      }
   }
}

use std::fs;
fn main() {
   match fs::read_dir(".") {
      Err(e) => println!("{}", e),
      Ok(readdir) => for r_dirent in readdir {
         match r_dirent {
            Err(e) => println!("{}", e),
            Ok(dirent) => {
               let name = dirent.file_name();
               // std::ffi::OsString` doesn't implement `std::fmt::Display
               println!("{}", name);
            }
         }
      }
   }
}

use std::fs;
fn main() {
   match fs::read_dir(".") {
      Err(e) => println!("{}", e),
      Ok(readdir) => for r_dirent in readdir {
         match r_dirent {
            Err(e) => println!("{}", e),
            Ok(dirent) => {
               let pathbuf = dirent.path();
               // std::path::PathBuf doesn't implement std::fmt::Display
               println!("{}", pathbuf);
            }
         }
      }
   }
}
