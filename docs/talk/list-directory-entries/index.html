Thu May 21 2020 

use std::{fs, io};
fn main() -> io::Result<()> {
   for o1 in fs::read_dir(".")? {
      let o2 = o1?.path();
      let s1 = o2.display();
      println!("{}", s1);
   }
   Ok(())
}

use std::fs;
fn main() {
   match fs::read_dir(".") {
      Err(e) => println!("{}", e),
      Ok(readdir) => for r_dirent in readdir {
         match r_dirent {
            Err(e) => println!("{}", e),
            Ok(dirent) => {
               match dirent.path().to_str() {
                  None => println!("to_str"),
                  Some(s1) => println!("{}", s1)
               }
            }
         }
      }
   }
}
