"a ""b"" c"

Dart

main(a) {
   var s = a[0];
   print(s == 'a "b');
}

Go

package main
import "os"
func main() {
   s := os.Args[1]
   println(s == `a "b`)
}

   let a: Vec<String> = env::args().collect();
   let s = &a[1];
   println!("{}", s == r#"a "b"#);
