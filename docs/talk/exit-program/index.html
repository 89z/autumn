Talk / Exit program

Sun Oct 11 2020

Rust needs a single line way to exit a program. With Go, you have Log.Fatal:

package main

import (
   "log"
   "os"
)

func main() {
   o, e := os.Stat("a.go")
   if e != nil {
      log.Fatal(e)
   }
   n := o.Size()
   println(n)
}

This works for a Result:

use std::{
   fs,
   io
};

fn main() -> io::Result<()> {
   let o = fs::metadata("a.rs")?;
   let n = o.len();
   println!("{}", n);
   Ok(())
}

This works for Option:

use std::char;

fn main() -> Result<(), u32> {
   let n = 10;
   let c = char::from_digit(n, 16).ok_or(n)?;
   println!("{}", c);
   Ok(())
}

but what if you have both in the same function? Need something that works for numbers as well. Go has an option:

package main

import (
   "fmt"
   "log"
)

func main() {
   e := fmt.Errorf("%v", 9)
   if e != nil {
      log.Fatal(e)
   }
}

Can we use Err(format!()) for everything? Yes we can:

use std::{
   char,
   error::Error
};

fn main() -> Result<(), Box<dyn Error>> {
   let n = 10;
   let c = char::from_digit(n, 16).ok_or(format!("{}", n))?;
   println!("{}", c);
   Ok(())
}

We can even do better:

use std::char;

fn main() -> Result<(), String> {
   let n = 10;
   let c = char::from_digit(n, 16).ok_or(format!("{}", n))?;
   println!("{}", c);
   Ok(())
}

but then what about traditional errors? Yep:

use std::fs;

fn main() -> Result<(), String> {
   let o = fs::metadata("a.rs").map_err(|e| format!("{}", e))?;
   println!("{}", o.len());
   Ok(())
}

Now lets combine:

use std::{
   char,
   error::Error,
   fs
};

fn main() -> Result<(), Box<dyn Error>> {
   // example 1
   let o = fs::metadata("a.rs")?;
   // example 2
   let n = 10;
   let c = char::from_digit(n, 16).ok_or(format!("{}", n))?;
   // print
   println!("{} {}", o.len(), c);
   Ok(())
}

If we dont like the long line, we can do this:

use std::{
   char,
   error::Error
};

fn main() -> Result<(), Box<dyn Error>> {
   let n = 10;
   let c = match char::from_digit(n, 16) {
      Some(v) => v,
      None => Err(format!("{}", n))?
   };
   println!("{}", c);
   Ok(())
}
