Talk:Reduce array

Mon Jan 1 0001

If you have some code like this:

<?php
$a = ['May', 'June'];
$f = fn ($s_acc, $s_cur) => $s_acc . $s_cur;
$s = array_reduce($a, $f);
var_dump($s);

You can rewrite it like this:

<?php
$a = ['May', 'June'];
$s_acc = '';

foreach ($a as $s_cur) {
   $s_acc .= $s_cur;
}

var_dump($s_acc);

Others:

package main

func reduce(s_acc, s_cur string) string {
   return s_acc + s_cur
}

func main() {
   a_src := []string{"May", "June"}
   s_acc := ""
   for n_idx := range a_src {
      s_cur := a_src[n_idx]
      s_acc = reduce(s_acc, s_cur)
   }
   println(s_acc)
}

package main

func main() {
   a_src := []string{"May", "June"}
   s_acc := ""
   for n_idx := range a_src {
      s_acc += a_src[n_idx]
   }
   println(s_acc)
}
