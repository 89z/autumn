Talk:Convert radix

Sat May 2 21:28:16 2020

If you percent encode everything from U+0021 to U+007E, then you have:

0-9
a-z
A-Z
-
.
_

which is 65 characters. Omit the . and that is 64.

C

#include <stdio.h>
// head
char *radix64(int n1) {
   char *s_safe = "0123456789abcdefghijklmnopqrstuvwxyz"
   "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
   static char s1[7];
   int n2;
   for (n2 = 0; n1 > 0; n2++) {
      s1[n2] = s_safe[n1 & 63];
      n1 >>= 6;
   }
   s1[n2] = 0;
   return s1;
}
// body
int main() {
   char *s1 = radix64(1588337932);
   puts(s1);
}

#include <stdio.h>
// head
char *radix64(long n1) {
   char *s_safe = "0123456789abcdefghijklmnopqrstuvwxyz"
   "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
   static char s1[7];
   char *s2 = s1;
   int n2;
   for (n2 = 0; n2 < 6; n2++) {
      if (n1 == 0) {
         *s2 = 0;
      }
      *s2++ = s_safe[n1 & 63];
      n1 >>= 6;
   }
   return s1;
}
// body
int main() {
   char *s1 = radix64(1588337932);
   puts(s1);
}
