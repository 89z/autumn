Talk:Slice a string

Tue May 5 2020 
s1 = 'Sunday'
# one from start
s2 = s1[0]
# two from start
s3 = s1[:2]
# two from middle
s4 = s1[1:3]
# from middle to end
s5 = s1[1:]
# one from end
s6 = s1[-1]
# two from end
s7 = s1[-2:]
# print
print(s2 == 'S', s3 == 'Su', s4 == 'un', s5 == 'unday', s6 == 'y', s7 == 'ay')
