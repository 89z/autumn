Talk:Slice a string

Tue May 5 2020

We want one character from start:

let s1 = 'Sunday';
let s2 = s1.charAt();
console.log(s2);

We want two characters from start:

package main
func main() {
   s1 := "Sunday"
   s2 := s1[:2]
   println(s2)
}

We want one character from middle:

package main
import "fmt"
func main() {
   s1 := "Sunday"
   s2 := s1[1:1]
   fmt.Printf("%q\n", s2) // ""
}

We want one character from end:

package main
func main() {
   s1 := "Sunday"
   s2 := s1[-1:]
   println(s2) // invalid slice index -1 (index must be non-negative)
}

We want two characters from end:

let s1 = 'Sunday';
let s2 = s1.slice(-2);
console.log(s2 == 'ay');

Result:

package main
func main() {
   s1 := "Sunday"
   // start one
   s2 := s1[:1]
   // start two
   s3 := s1[:2]
   // mid one
   s4 := s1[1:2]
   // end one
   s5 := s1[len(s1) - 1:]
   // end two
   s6 := s1[len(s1) - 2:]
   // print
   println(s2 == "S", s3 == "Su", s4 == "u", s5 == "y", s6 == "ay")
}
