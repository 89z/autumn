Go

package main
import (
   "image"
   "image/jpeg"
   "os"
)
func main() {
   // files
   f1, _ := os.Open("a.jpg")
   f2, _ := os.Create("b.jpg")
   // rectangle
   r1 := image.Rect(800, 800, 1400, 1400)
   // images
   i1, _ := jpeg.Decode(f1)
   i2 := i1.(*image.YCbCr).SubImage(r1)
   // write
   jpeg.Encode(f2, i2, nil)
}

package main
import (
   "image"
   "image/draw"
   "image/jpeg"
   "os"
)
func main() {
   // files
   f1, _ := os.Open("a.jpg")
   f2, _ := os.Create("b.jpg")
   // rectangle
   r1 := image.Rect(800, 800, 1400, 1400)
   // images
   i1, _ := jpeg.Decode(f1)
   i2 := image.NewRGBA(r1)
   // write
   draw.Draw(i2, r1, i1, r1.Min, draw.Src)
   jpeg.Encode(f2, i2, nil)
}

PHP

<?php
# files
$s1 = 'p80eaiz8t0n41.jpg';
$s2 = 'out.jpg';
# rectangle
$m1 = ['x' => 0, 'y' => 0, 'width' => 800, 'height' => 800];
# images
$r1 = imagecreatefromjpeg($s1);
$r2 = imagecrop($r1, $m1);
# write
imagejpeg($r2, $s2);

Warning, this requires php-gd package, which pulls in 23 other packages.
