function addYear(d) {
   let n = d.getFullYear();
   d.setFullYear(n + 1);
}

let d = new Date;
addYear(d);
console.log(d);

https://developer.mozilla.org/docs/Web/JavaScript/Reference/Global_Objects/Date/setFullYear

function addMonth(d) {
   let n = d.getMonth();
   d.setMonth(n + 1);
}

let d = new Date;
addMonth(d);
console.log(d);

https://developer.mozilla.org/docs/Web/JavaScript/Reference/Global_Objects/Date/setMonth
