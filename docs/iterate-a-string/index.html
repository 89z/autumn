Iterate a string

Wed Mar 4 18:45:59 2020 iterate string

Go

package main
import (
   "bufio"
   "strings"
)
func main() {
   s1 := `Sunday
Monday`
   o1 := bufio.NewScanner(strings.NewReader(s1))
   for o1.Scan() {
      println(o1.Text())
   }
}

PHP

<?php
$s1 = 'Sunday
Monday';
$s2 = strtok($s1, "\n");
while ($s2 !== false) {
   var_dump($s2);
   $s2 = strtok("\n");
}
