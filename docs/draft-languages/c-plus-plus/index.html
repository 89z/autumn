Regarding booleans, it is not ideal that numbers are printed by default:

#include <iostream>
int main() {
   auto b1 = true;
   // example 1
   std::cout << b1 << std::endl;
   // example 2
   std::cout << std::boolalpha << b1 << std::endl;
   // example 3
   std::cout << std::noboolalpha << b1 << std::endl;
}

but to be fair, other languages have similar issues. Here is Go:

package main
import "fmt"
func main() {   
   a1 := []string{"Sun", "Mon Tue"}
   // example 1
   fmt.Println(a1)
   // example 2
   fmt.Printf("%q\n", a1)
}

Here is creating a map:

#include <iostream>
#include <map>
#include <string>
int main() {
   std::map<std::string, int> m1 = {{"Sun", 10}, {"Mon", 11}};
   for (auto &[s1, n1]: m1) {
      std::cout << s1 << "," << n1 << std::endl;
   }
}
