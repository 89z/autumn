use chrono::{
   DateTime,
   offset
};

fn main() -> Result<(), chrono::ParseError> {
   let s = "2020-12-31T01:02:31Z";
   let o: DateTime<offset::Utc> = s.parse()?;
   println!("{}", o);
   Ok(())
}

use chrono::{
   TimeZone,
   Utc
};

fn main() -> Result<(), chrono::ParseError> {
   let s= "2020-12-31 01:02:31";
   let o  = Utc.datetime_from_str(s, "%Y-%m-%d %H:%M:%S")?;
   println!("{:?}", o);
   Ok(())
}
