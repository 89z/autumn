HTTP GET / Dart

Thu Apr 1 2021 network 

import 'dart:io';

void main() {
   var uri = Uri.parse("https://speedtest.lax.hivelocity.net");
   var http = new HttpClient();
   http.getUrl(uri).then(
      (req) => req.close()
   ).then(
      (res) => res.pipe(stdout)
   );
   http.close();
}

import 'dart:io';

void main() async {
   var uri = Uri.parse("https://speedtest.lax.hivelocity.net");
   var http = new HttpClient();
   var req = await http.getUrl(uri);
   var res = await req.close();
   res.pipe(stdout);
   http.close();
}
