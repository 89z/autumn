HTTP GET

Sun May 24 2020 network Talk:HTTP GET

Go

package main
import (
   "io/ioutil"
   "log"
   "net/http"
)
func main() {
   s := "http://speedtest.lax.hivelocity.net"
   o, e := http.Get(s)
   if e != nil {
      log.Fatal(e)
   }
   y, e := ioutil.ReadAll(o.Body)
   if e != nil {
      log.Fatal(e)
   }
   log.Printf("%s", y)
}

Rust

use curl::easy::{Easy, WriteError};
use std::error::Error;
use std::io::{stdout, Write};
fn main() -> Result<(), Box<dyn Error>> {
   let mut o = Easy::new();
   o.url("http://speedtest.lax.hivelocity.net")?;
   o.write_function(|y|
      stdout().write(y).or(Err(WriteError::Pause))
   )?;
   o.perform()?;
   Ok(())
}
