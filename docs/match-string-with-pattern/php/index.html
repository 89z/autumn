Thu Dec 17 2020 string 

<?php

function pattern(string $pattern, string $subject): bool {
   return preg_match($pattern, $subject) == 1;
}

$b = pattern('/..n/', 'Sunday Monday');
var_dump($b);

<?php

function pattern(string $pattern, string $subject): array|bool {
   $n = preg_match($pattern, $subject, $a);
   if ($n != 1) {
      return false;
   }
   return $a;
}

$a = pattern('/..n/', 'Sunday Monday');
var_dump($a);
