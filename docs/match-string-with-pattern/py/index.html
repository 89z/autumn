Match string with pattern / Python

Mon Jun 14 2021 string 
import re
s = 'South North'

# example 1
r = re.search('so', s, re.I)
print(r != None)

# example 2
r = re.search('(?i)so', s)
print(r != None)

https://docs.python.org/library/re.html#re.I

import re
s = 'south north'
r = re.search('^s', s)
print(r != None)
