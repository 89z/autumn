Go

package main
func main() {
   // example 1
   n1 := 10
   for true {
      println(n1)
      if n1 == 19 {
         break
      }
      n1++
   }
   // example 2
   n2 := 10
   for {
      println(n2)
      if n2 == 19 {
         break
      }
      n2++
   }
}

JavaScript

// example 1
let n1 = 10;
do {
   console.log(n1);
   n1++;
} while (n1 < 20);
// example 2
let n2 = 10;
while (true) {
   console.log(n2);
   if (n2 == 19) {
      break;
   }
   n2++;
}
// example 3
let n3 = 10;
for (;;) {
   console.log(n3);
   if (n3 == 19) {
      break;
   }
   n3++;
}

Python

# example 1
n1 = 10
while True:
   print(n1)
   if n1 == 19:
      break
   n1 += 1
# example 2
def f1(n1):
   print(n1)
   if n1 >= 19:
      return
   f1(n1 + 1)
f1(10)

Exclude range as it is ambiguous.
