import std.stdio;
void main() {
   auto n = 10;
   do {

      n++;
   } while (n < 20);
}

Go

package main
func main() {
   var n int
   // example 1
   n = 10
   for true {
      println(n)
      if n == 19 {
         break
      }
      n++
   }
   // example 2
   n = 10
   for {
      println(n)
      if n == 19 {
         break
      }
      n++
   }
}

JavaScript

let n;
// example 1
n = 10;
do {
   console.log(n);
   n++;
} while (n < 20);
// example 2
n = 10;
while (true) {
   console.log(n);
   if (n == 19) {
      break;
   }
   n++;
}
// example 3
n = 10;
for (;;) {
   console.log(n);
   if (n == 19) {
      break;
   }
   n++;
}
