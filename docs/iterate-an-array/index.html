Iterate an array

Wed Mar 4 18:45:59 2020 iterate array

Go

package main
func main() {
   a1 := []string{"Sun", "Mon"}
   // example 1
   for n1, s1 := range a1 {
      println(n1, s1)
   }
   // example 2
   for n1 := range a1 {
      println(n1)
   }
   // example 3
   for _, s1 := range a1 {
      println(s1)
   }
}

https://golang.org/ref/spec#For_range

JavaScript

let a1 = ['Sun', 'Mon'];
// example 1
for (let n1 = 0; n1 < a1.length; n1++) {
   console.log(a1[n1]);
}
// example 2
for (let n1 in a1) {
   console.log(a1[n1]);
}
// example 3
for (let s1 of a1) {
   console.log(s1);
}
// example 4
a1.forEach(s1 => console.log(s1));

PHP

<?php
$a1 = ['Sun', 'Mon'];
# example 1
foreach ($a1 as $s1) {
   var_dump($s1);
}
# example 2
for ($n1 = 0; $n1 < count($a1); $n1++) {
   var_dump($a1[$n1]);
}
# example 3
for (reset($a1); key($a1) !== null; next($a1)) {
   var_dump(current($a1));
}
# example 4
array_walk($a1, function ($s1) {
   var_dump($s1);
});
