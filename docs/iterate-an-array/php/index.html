Sun Nov 1 2020 iterate array 
<?php
$a = ['May', 'June'];
# example 1
for ($n = 0; $n < count($a); $n++) {
   echo $n, "\t", $a[$n], "\n";
}
# example 2
for ($n = 0; key_exists($n, $a); $n++) {
   echo $n, "\t", $a[$n], "\n";
}

   echo $s, "\n";

<?php
$a = ['May', 'June'];
# example 1
$n = 0;
while ($n < count($a)) {
   echo $n, "\t", $a[$n], "\n";
   $n++;
}
# example 2
$n = 0;
while (key_exists($n, $a)) {
   echo $n, "\t", $a[$n], "\n";
   $n++;
}

<?php
$a = ['May', 'June'];

while (true) {
   $n = key($a);
   if ($n === null) {
      break;
   }
   $s = current($a);
   echo $n, "\t", $s, "\n";
   next($a);
}
