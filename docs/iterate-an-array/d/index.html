Iterate an array / D

Sun Nov 1 2020 iterate array 
import std.stdio;

void main() {
   auto a = ["May", "June"];
   // example 1
   foreach (s; a) {
      s.writeln;
   }
   // example 2
   foreach (n, s; a) {
      writeln(n, ',', s);
   }
}
