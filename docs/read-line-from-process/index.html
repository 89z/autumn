Read line from process

Wed Mar 25 10:17:37 2020 input-output

Go

package main
import (
   "bufio"
   "os/exec"
)
func main() {
   // head
   o1 := exec.Command("ag", "-V")
   o2, _ := o1.StdoutPipe()
   o1.Start()
   // body
   o3 := bufio.NewScanner(o2)
   o3.Scan()
   s1 := o3.Text()
   println(s1)
}

Python

import subprocess
a1 = ['ag', '-V']
o1 = subprocess.Popen(a1, stdout=subprocess.PIPE, text=True)
s1 = o1.stdout.readline()
print(s1)
