package main
import "math/bits"

func divMod(d, e uint32) (uint32, uint32) {
   return bits.Div32(0, d, e)
}

func main() {
   d, m := divMod(0xFFFF_FFFF, 2)
   println(d == 0x7FFF_FFFF, m == 1)
}

package main

func main() {
   { // natural int
      n := 7 / 2
      println(n == 3)
   }
   { // natural float
      n := 7.0 / 2
      println(n == 3.5)
   }
   { // force int
      n := int(7.0) / 2
      println(n == 3)
   }
   { // force float
      n := float64(7) / 2
      println(n == 3.5)
   }
}
