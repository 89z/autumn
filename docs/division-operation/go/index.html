package main
import "math/bits"

func divMod(d, e uint32) (uint32, uint32) {
   return bits.Div32(0, d, e)
}

func main() {
   d, m := divMod(0xFFFF_FFFF, 2)
   println(d == 0x7FFF_FFFF, m == 1)
}

   { // example 1
      n := f / 2
      println(n == 3.75)
   }
   { // example 2
      n := int(f) / 2
      println(n == 3)
   }
   { // example 3
      n := i / 2
      println(n == 3)
   }
   { // example 4
      n := float64(i) / 2

}
