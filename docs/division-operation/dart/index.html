void main() {
   var f = 7.5, i = 7;
   // example 1
   var n1 = f / 2;
   // example 2
   var n2 = (f / 2).toInt();
   // example 3
   var n3 = (i / 2).toInt();
   // example 4
   var n4 = i / 2;
   // print
   print(n1 == 3.75 && n2 == 3 && n3 == 3 && n4 == 3.5);
}
