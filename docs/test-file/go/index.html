package main
import "os"

func main() {
   o, e := os.Stat("a.go")
   if os.IsNotExist(e) || ! o.Mode().IsRegular() {
      os.Exit(1)
   }
}

https://golang.org/pkg/os#FileMode.IsRegular

Note that for existing files, these are four times slower:

package main
import "os"

func main() {
   o, e := os.Open("a.go")
   if os.IsNotExist(e) {
      os.Exit(1)
   }
   o.Close()
}

https://golang.org/pkg/os#Open

package main
import "os"

func main() {
   o, e := os.OpenFile("d.go", os.O_CREATE | os.O_EXCL, os.ModePerm)
   if os.IsExist(e) {
      os.Exit(1)
   }
   o.Close()
}
