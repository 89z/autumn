Number to date string

Wed Jun 10 2020 time

Go

package main
import "time"
func main() {
   var n int64 = 365 * 24 * 60 * 60
   var o = time.Unix(n, 0)
   var s = o.String()
   println(s)
}

https://golang.org/pkg/time#Time.String

JavaScript

let n = 365 * 24 * 60 * 60;
let o = new Date(n * 1000);
let s1 = o.toLocaleString(0, {weekday: 'short'});
let s2 = o.toLocaleString(0, {month: 'short', day: 'numeric'});
let s3 = o.toLocaleString(0, {year: 'numeric'});
console.log(s1, s2, s3);

https://developer.mozilla.org/Web/JavaScript/Reference/Global_Objects/Date/toLocaleString

PHP

<?php
$n = 365 * 24 * 60 * 60;
$s = strftime('%F', $n);
var_dump($s);

https://php.net/function.strftime

Python

from datetime import date
n = 365 * 24 * 60 * 60
o = date.fromtimestamp(n)
# example 1
s1 = o.isoformat()
# example 2
s2 = o.strftime('%F')
# print
print(s1, s2)
  • https://docs.python.org/library/datetime.html#datetime.date.isoformat
  • https://docs.python.org/library/datetime.html#datetime.date.strftime
