D

import std.process, std.stdio;
void main() {
   const s1 = "cal";
   const s2 = executeShell(s1).output;
   write(s2);
}

Go

package main
import (
   "fmt"
   "os/exec"
)
func main() {
   t1 := exec.Command("cal")
   a1, _ := t1.Output()
   fmt.Printf("%s", a1)
}

https://golang.org/pkg/os/exec#Cmd.Output

# print
