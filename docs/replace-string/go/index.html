package main
import "regexp"

func main() {
   s := "west? east.txt"
   s = regexp.MustCompile(`["*/:<>?\|]`).ReplaceAllString(s, "")
   println(s == "west east.txt")
}

package main
import "strings"

func main() {
   s := "west? east.txt"
   var a []string
   for _, r := range `"*/:<>?\|` {
      a = append(a, string(r), "")
   }
   s = strings.NewReplacer(a...).Replace(s)
   println(s == "west east.txt")
}

package main
import "strings"

func main() {
   s := "west? east.txt"
   for _, r := range `"*/:<>?\|` {
      s = strings.ReplaceAll(s, string(r), "")
   }
   println(s == "west east.txt")
}

package main
import "strings"

func main() {
   var b strings.Builder
   for _, r := range "west? east.txt" {
      if ! strings.ContainsRune(`"*/:<>?\|`, r) {
         b.WriteRune(r)
      }
   }
   println(b.String() == "west east.txt")
}

package main
import "strings"

func main() {
   s := "west? east.txt"
   s = strings.Map(func(r rune) rune {
      switch {
      case strings.ContainsRune(`"*/:<>?\|`, r): return -1
      default: return r
      }
   }, s)
   println(s == "west east.txt")
}
