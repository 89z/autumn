package main

func FoldString(a []string, f func(string, string) string) string {
   s := ""
   for _, s1 := range a {
      s = f(s, s1)
   }
   return s
}

func main() {
   a := []string{"May", "June"}
   s := FoldString(a, func (s, s1 string) string {
      return s + s1
   })
   println(s == "MayJune")
}

package main

func FoldIntString(a []string, f func(int, string) int) int {
   n := 0
   for _, s := range a {
      n = f(n, s)
   }
   return n
}

func main() {
   a := []string{"May", "June"}
   n := FoldIntString(a, func (n int, s string) int {
      return n + len(s)
   })
   println(n == 7)
}
