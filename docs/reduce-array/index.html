Go

package main

import (
	"fmt"
)

func main() {
	const c = 5.5
	elements := []float64{2, 4, 9, 30}
	fmt.Println(elements)

	for i := range elements {
		elements[i] *= c
	}
	fmt.Println(elements)
}

<?php
$a = ['May', 'June'];

<?php
$a = ['May', 'June'];
$m = [];
$f = fn ($m_acc, $s_cur, $n_idx) => $m_acc + [$s_cur => $n_idx];

foreach ($a as $n => $s) {
   $m = $f($m, $s, $n);
}

print_r($m);
