String to integer / Go

Mon Sep 28 2020 number 

package main

import (
   "os"
   "strconv"
)

func main() {
   n, e := strconv.Atoi("10")
   if e != nil {
      os.Exit(1)
   }
   println(n == 10)
}

package main
import "fmt"

func main() {
   // example 1
   n1 := 0
   fmt.Sscanln("10", &n1)
   // example 2
   n2 := 0
   fmt.Sscan("10", &n2)
   // example 3
   n3 := 0
   fmt.Sscanf("10", "%v", &n3)
   // print
   println(n1 == 10, n2 == 10, n3 == 10)
}

package main

import (
   "os"
   "strconv"
)

func main() {
   n, e := strconv.ParseInt("11", 10, 0)
   if e != nil {
      os.Exit(1)
   }
   println(n == 11)
}
