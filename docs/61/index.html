Go

package main
import (
   "io/ioutil"
   "os"
)
func main() {
   a, e := ioutil.ReadDir(".")
   if e != nil {
      os.Exit(1)
   }
   for n, o := range a {
      s := o.Name()
      println(n, s)
   }
}

package main
import (
   "fmt"
   "log"
   "os"
)
func main() {
   o, e := os.Open(".")
   if e != nil {
      log.Fatal(e)
   }
   a, e := o.Readdirnames(0)
   if e != nil {
      log.Fatal(e)
   }
   fmt.Println(a)
}

package main
import (
   "os"
   "path/filepath"
)
func f(s string, o os.FileInfo, e error) error {
   if o.IsDir() {
      return e
   }
   println(s)
   return nil
}
func main() {
   filepath.Walk(".", f)
}

PHP

<?php
$o_iter = new FilesystemIterator('.');
foreach ($o_iter as $o_info) {
   echo $o_info->getPathname(), "\n";
}

<?php
$o_dir = new RecursiveDirectoryIterator('.');
$o_dir->setFlags(RecursiveDirectoryIterator::SKIP_DOTS);
$o_iter = new RecursiveIteratorIterator($o_dir);
foreach ($o_iter as $o_info) {
   echo $o_info->getPathname(), "\n";
}

<?php
$o_dir = new RecursiveDirectoryIterator('.');
$o_dir->setFlags(RecursiveDirectoryIterator::SKIP_DOTS);
$f_filter = fn ($o_info) => $o_info->getFilename() == '.git' ? false : true;
$o_filter = new RecursiveCallbackFilterIterator($o_dir, $f_filter);
$o_iter = new RecursiveIteratorIterator($o_filter);
foreach ($o_iter as $o_info) {
   echo $o_info->getPathname(), "\n";
}

Rust

use std::{fs, io};
fn main() -> io::Result<()> {
   for r_ent in fs::read_dir(".")? {
      let s_ent = r_ent?.path();
      println!("{:?}", s_ent);
   }
   Ok(())
}
