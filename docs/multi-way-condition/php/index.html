Multi-way condition / PHP

Fri Sep 11 2020 reference 
<?php
$n = 10;

switch ($n) {
case 12:
   $s = 'all';
   break;
case 11:
case 10:
   $s = 'some';
   break;
default:
   $s = 'none';
}

var_dump($s == 'some');
