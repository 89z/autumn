Multi-way condition

Fri Sep 11 2020 reference

C#

using System;

class Program {
   static void Main() {
      int n = 10;
      string s;
      switch (n) {
      case 12:
         s = "all";
         break;
      case 11:
      case 10:
         s = "some";
         break;
      default:
         s = "none";
         break;
      }
      Console.WriteLine(s == "some");
   }
}

Go

package main

func main() {
   n, s := 10, ""
   switch n {
   case 12:
      s = "all"
   case 11, 10:
      s = "some"
   default:
      s = "none"
   }
   println(s == "some")
}
