Multi-way condition

Sat Jun 27 2020 reference

Dart

main() {
   var s = 'Sun';
   switch (s) {
   case 'Fri':
      print(1);
      break;
   case 'Sat':
   case 'Sun':
      print(2);
      break;
   default:
      print(0);
   }
}

Go

package main
func main() {
   var s = "Sun"
   var n int
   switch s {
   case "Fri":
      n = 1
   case "Sat", "Sun":
      n = 2
   default:
      n = 0
   }
   println(n)
}

JavaScript

let n, s = 'Sun';
switch (s) {
case 'Fri':
   n = 1;
   break;
case 'Sat':
case 'Sun':
   n = 2;
   break;
default:
   n = 0;
}
console.log(n);
