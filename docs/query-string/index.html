Query string

Tue Jun 9 2020 network

Go

package main
import (
   "fmt"
   "log"
   "net/url"
)
func main() {
   // example 1
   s1 := "one=odd&two=even"
   m1, e := url.ParseQuery(s1)
   if e != nil {
      log.Fatal(e)
   }
   // example 2
   m2 := url.Values{
      "one": []string{"odd"},
      "two": []string{"even"},
   }
   s2 := m2.Encode()
   // print
   fmt.Println(m1, s2)
}

<?php

parse_str($s1, $m1);

import urllib.parse as o1
# example 1
s1 = 'one=odd&two=even'
m1 = o1.parse_qs(s1)
# example 2
m2 = {'one': 'odd', 'two': 'even'}
s2 = o1.urlencode(m2)
# print
