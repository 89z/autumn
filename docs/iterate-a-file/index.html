Autumn / Iterate a file

Fri Sep 11 2020

D

import std.stdio;

void main() {
   auto o = "a.txt".File;
   foreach (s; o.byLine) {
      s.writeln;
   }
}

https://dlang.org/library/std/stdio/file.by_line.html

Go

package main

import (
   "bufio"
   "os"
)

func main() {
   o, e := os.Open("a.txt")
   if e != nil {
      os.Exit(1)
   }
   o1 := bufio.NewScanner(o)
   for o1.Scan() {
      s := o1.Text()
      println(s)
   }
}

PHP

<?php
$r = fopen('a.txt', 'r');

while (true) {
   $s = fgets($r);
   if (feof($r)) {
      break;
   }
   var_dump($s);
}

feof is triggered after fgets. Further, trim expects a string, so we cannot use it until we are certain that we have a string.

https://php.net/function.feof

Python

for s in open('a.txt'):
   print(s, end='')

https://docs.python.org/reference/compound_stmts.html#for

Rust

use std::fs::File;
use std::io::{BufRead, BufReader, Error};

fn main() -> Result<(), Error> {
   let r_file = File::open("a.txt")?;
   for r_line in BufReader::new(r_file).lines() {
      let s_line = r_line?;
      println!("{}", s_line);
   }
   Ok(())
}

https://doc.rust-lang.org/std/io/trait.BufRead.html#method.lines
