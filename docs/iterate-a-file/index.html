Iterate a file

Tue May 19 2020 input-output iterate

PHP

<?php
$r1 = fopen('a.txt', 'r');
while (true) {
   $s1 = fgets($r1);
   if (feof($r1)) {
      break;
   }
   var_dump($s1);
}

feof is triggered after fgets. Further, trim expects a string, so we cannot use it until we are certain that we have a string.

https://php.net/function.feof

Python

for s1 in open('a.txt'):
   print(s1, end='')

https://docs.python.org/reference/compound_stmts.html#for

Rust

use std::fs::File;
use std::io::{BufRead,BufReader};
fn main() {
   match File::open("a.txt") {
      Err(e) => println!("{}", e),
      Ok(o1) => for o2 in BufReader::new(o1).lines() {
         match o2 {
            Err(e) => println!("{}", e),
            Ok(s1) => println!("{}", s1)
         }
      }
   }
}
