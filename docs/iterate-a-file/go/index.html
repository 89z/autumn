Iterate a file / Go

Mon Oct 5 2020 input-output 
package main

import (
   "bufio"
   "os"
)

func main() {
   file, e := os.Open("a.go")
   if e != nil {
      panci(e)
   }
   defer file.Close()
   buf := bufio.NewScanner(file)
   for buf.Scan() {
      println(buf.Text())
   }
}
