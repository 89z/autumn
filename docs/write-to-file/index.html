Write to file

Sun Jun 28 2020 input-output

D

import std.stdio;
void main() {
   auto o = File("a.txt", "w");
   o.writeln("Sunday");
}

https://dlang.org/library/std/stdio/file.writeln.html

Dart

import 'dart:io';
main() {
   var o = new File('a.txt');
   o.writeAsStringSync('Sunday\n');
}

https://api.dart.dev/dart-io/File-class.html

Go

package main
import (
   "io"
   "log"
   "os"
   "strings"
)
func main() {
   o1 := strings.NewReader("Sunday\n")
   o2, e := os.Create("a.txt")
   if e != nil {
      log.Fatal(e)
   }
   io.Copy(o2, o1)
}

package main
import (
   "fmt"
   "log"
   "os"
)
func main() {
   o, e := os.Create("a.txt")
   if e != nil {
      log.Fatal(e)
   }
   fmt.Fprintln(o, "Sunday")
}

package main
import (
   "log"
   "os"
)
func main() {
   o, e := os.Create("a.txt")
   if e != nil {
      log.Fatal(e)
   }
   o.WriteString("Sunday\n")
}

package main
import (
   "log"
   "os"
   "strings"
)
func main() {
   o1 := strings.NewReader("Sunday\n")
   o2, e := os.Create("a.txt")
   if e != nil {
      log.Fatal(e)
   }
   o1.WriteTo(o2)
}
