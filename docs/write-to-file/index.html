Go

package main
import "fmt"
import "os"
func main() {
   o1, _ := os.Create("a.txt")
   fmt.Fprintln(o1, "Sunday")
}

https://golang.org/pkg/fmt#Fprintln

package main
import "io"
import "os"
import "strings"
func main() {
   o1 := strings.NewReader("Sunday\n")
   o2, _ := os.Create("a.txt")
   io.Copy(o2, o1)
}

https://golang.org/pkg/io#Copy

package main
import "os"
func main() {
   o1, _ := os.Create("a.txt")
   o1.WriteString("Sunday\n")
}

https://golang.org/pkg/os#File.WriteString

package main
import "os"
import "strings"
func main() {
   o1 := strings.NewReader("Sunday\n")
   o2, _ := os.Create("a.txt")
   o1.WriteTo(o2)
}

https://golang.org/pkg/strings#Reader.WriteTo
