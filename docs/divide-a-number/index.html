Divide a number

Sat May 9 2020 number

Go

package main
func main() {
   n1 := 19
   n2 := 10
   // example 1
   n3 := n1 / n2
   // example 2
   n4 := float64(n1) / float64(n2)
   // print
   println(n3, n4)
}

JavaScript

// float
let n1 = 19 / 10;
// remainder
let n2 = 19 % 10;
// print
console.log(n1, n2);

PHP

<?php
# float
$n1 = 19 / 10;
# remainder
$n2 = 19 % 10;
# truncate
$n3 = intdiv(19, 10);
# print
var_dump($n1, $n2, $n3);

<?php
