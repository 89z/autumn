Go

package main
func main() {
   n1 := 19
   n2 := 10
   // example 1
   n3 := n1 / n2
   // example 2
   n4 := float64(n1) / float64(n2)
   // example 3
   n5 := n1 % n2
   // print
   println(n3, n4, n5)
}

package main
import "math"
func main() {
   // example 1
   n1 := math.Mod(19, 10)
   // example 2
   n2 := math.Remainder(19, 10)
   // print
   println(n1 == 9, n2 == -1)
}
