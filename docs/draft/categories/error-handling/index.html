Error handling

Sun May 24 2020 draft

Go

package main
import (
   "log"
   "strconv"
)
func main() {
   n, e := strconv.Atoi("10")
   if e != nil {
      log.Fatal(e)
   }
   println(n)
}

Rust

use curl::easy::{Easy, WriteError};
use std::error::Error;
use std::io::{stdout, Write};
fn main() -> Result<(), Box<dyn Error>> {
   let mut o = Easy::new();
   o.url("http://speedtest.lax.hivelocity.net")?;
   o.write_function(|y|
      stdout().write(y).or(Err(WriteError::Pause))
   )?;
   o.perform()?;
   Ok(())
}
