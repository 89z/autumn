Error handling

Sat May 23 2020 draft

Go

package main
import (
   "log"
   "strconv"
)
func main() {
   n, e := strconv.Atoi("10")
   if e != nil {
      log.Fatal(e)
   }
   println(n)
}

Rust

use std::env;
use std::error::Error;
fn main() -> Result<(), Box<dyn Error>> {
   // test 1
   let s1 = env::var("APPDATA")?;
   println!("{}", s1);
   // test 2
   let s2 = env::var("DATAAPP")?;
   println!("{}", s2);
   // test 3
   println!("Sunday");
   Ok(())
}

use std::collections::HashMap;
use std::error::Error;
fn main() -> Result<(), Box<dyn Error>> {
   let mut m = HashMap::new();
   m.insert("Sunday", 10);
   // std::error::Error is not implemented for std::option::NoneError
   let n = m.get("Sunday")?;
   println!("{}", n);
   Ok(())
}

use std::collections::HashMap;
fn main() {
   let mut m = HashMap::new();
   m.insert("Sunday", 10);
   if let Some(n) = m.get("Sunday") {
      println!("{}", n);
   }
}

use std::collections::HashMap;
use std::error::Error;
fn main() -> Result<(), Box<dyn Error>> {
   let mut m = HashMap::new();
   m.insert("Sunday", 10);
   let n = m.get("Sunday").ok_or("None")?;
   println!("{}", n);
   Ok(())
}

use std::collections::HashMap;
fn main() {
   let mut m = HashMap::new();
   m.insert("Sunday", 10);
   match m.get("Sunday") {
      None => return,
      Some(n) => println!("{}", n)
   }
}

use std::collections::HashMap;
fn main() {
   let mut m1 = HashMap::new();
   m1.insert("Sunday", 10);
   let n1 = match m1.get("Sunday") {
      None => {
         println!("None");
         return;
      },
      Some(n) => n
   };
   println!("{}", n1);
}

use std::collections::HashMap;
fn main() -> Result<(), ()> {
   let mut m1 = HashMap::new();
   m1.insert("Sunday", 10);
   // test 1
   let n1 = match m1.get("Sunday") {
      None => return Err(()),
      Some(n) => n
   };
   println!("{}", n1);
   // test 2
   let n2 = match m1.get("Monday") {
      None => return Err(()),
      Some(n) => n
   };
   println!("{}", n2);
   // test 3
   println!("Ok");
   Ok(())
}
