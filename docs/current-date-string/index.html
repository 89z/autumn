Go

package main
import "time"

func main() {
   o := time.Now()
   // example 1
   s := o.String()
   // example 2
   s2 := o.Format(time.RFC3339)
   // print
   print(s, "\n", s2, "\n")
}

package main

import (
   "fmt"
   "os"
   "time"
)

func main() {
   y, e := time.Now().MarshalText()
   if e != nil {
      os.Exit(1)
   }
   fmt.Printf("%s\n", y)
}
