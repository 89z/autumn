$s = 'June';

$n = strpos($s, 'Ju');
var_dump($n === 0);
# example 2
$n = strpos($s, 'un');

# example 3
$n = stripos($s, 'ju');
var_dump($n === 0);
# example 4

var_dump($s4 == 'J');
