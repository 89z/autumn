Go

package main
import (
   "fmt"
   "net/url"
)
func main() {
   s1 := "http://sun.com/mon?tue=10"
   m1, _ := url.Parse(s1)
   fmt.Printf("%#v\n", m1)
}

https://golang.org/pkg/net/url#Parse

JavaScript

// example 1
let s1 = 'http://sun.com/mon?tue=10';
let o1 = new URL(s1);
// example 2
let s2 = 'http://sun.com';
let s3 = 'mon?tue=10';

PHP

<?php
$s1 = 'https://example.com/sun?mon=10';
# example 1
$m1 = parse_url($s1);
# example 2
$s2 = parse_url($s1, PHP_URL_PATH);
# example 3
$s3 = parse_url($s1, PHP_URL_QUERY);
# example 4
$s4 = parse_url($s1)['path'];
# example 5
$s5 = parse_url($s1)['query'];
# print
var_dump($m1, $s2, $s3, $s4, $s5);

https://php.net/function.parse-url
