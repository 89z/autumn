Go

package main
import (
   "fmt"
   "net/url"
)
func main() {
   s1 := "http://sun.com/mon?tue=10"
   m1, _ := url.Parse(s1)
   fmt.Printf("%#v\n", m1)
}

JavaScript

PHP

<?php
$s1 = 'https://example.com/sun?mon=10';
# example 1
$s2 = parse_url($s1, PHP_URL_HOST);
# example 2
$s3 = parse_url($s1, PHP_URL_PATH);
# example 3
$s4 = parse_url($s1, PHP_URL_QUERY);
# print
var_dump($s2, $s3, $s4);

<?php
$s1 = 'https://example.com/sun?mon=10';
$m1 = parse_url($s1);
# example 1
$s2 = $m1['host'];
# example 2
$s3 = $m1['path'];
# example 3
$s4 = $m1['query'];
# print
var_dump($s2, $s3, $s4);

from urllib import parse
# example 1
s1 = 'https://example.com/one?two=even'
o1 = parse.urlparse(s1)
print(o1)
# example 2
s2 = o1.geturl()
print(s2)
