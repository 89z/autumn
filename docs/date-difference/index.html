Date difference

Tue Jun 9 2020 time

Go

package main
import (
   "fmt"
   "time"
)
func main() {
   o1 := time.Date(2019, time.December, 31, 0, 0, 0, 0, time.UTC)
   o2 := time.Now()
   o3 := o2.Sub(o1)
   fmt.Println(o3)
}

package main
import (
   "fmt"
   "log"
   "time"
)
func main() {
   o1, e := time.Parse(time.RFC3339, "2019-12-31T00:00:00Z")
   if e != nil {
      log.Fatal(e)
   }
   o2 := time.Now()
   o3 := o2.Sub(o1)
   fmt.Println(o3)
}
