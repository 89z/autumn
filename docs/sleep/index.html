Sleep

D

import core.thread, std.stdio;
void main() {
   "Sunday".writeln;
   Thread.sleep(2.seconds);
   "Monday".writeln;
}

Dart

import 'dart:io';
main() {
   print('Sunday');
   sleep(const Duration(seconds: 2));
   print('Monday');
}
