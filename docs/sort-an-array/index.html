Sort an array

Thu Jul 16 2020 array

C++

#include <iostream>
#include <vector>
int main() {
   std::vector<std::string> a = {"BBBB", "AA", "CCC"};
   std::sort(a.begin(), a.end());
   for (auto s: a) {
      std::cout << s << std::endl;
   }
}

Go

package main
import (
   "fmt"
   "sort"
)
func main() {
   a := []string{"BBBB", "AA", "CCC"}
   f := func(n1, n2 int) bool {
      return len(a[n1]) < len(a[n2])
   }
   sort.Slice(a, f)
   fmt.Println(a)
}

<?php
