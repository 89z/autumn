Sort an array

Sat May 2 2020 array

Go

package main
import (
   "fmt"
   "sort"
)
func main() {
   // example 1
   a1 := []int{20, 10}
   a2 := sort.IntSlice(a1)
   // example 2
   a3 := []string{"Monday", "Tue"}
   sort.Slice(a3, func(n1, n2 int) bool {
      return len(a3[n1]) < len(a3[n2])
   })
   // print
   fmt.Println(a2, a3)
}

PHP

<?php
# example 1
$a1 = ['Sun', 'Mon'];
sort($a1);
# example 2
$a2 = ['Mon', 'Tue'];
rsort($a2);
# example 3
$a3 = ['Monday', 'Tue'];
array_multisort(array_map('strlen', $a3), $a3);
# example 4
$a4 = ['Monday', 'Tue'];
usort($a4, function ($s1, $s2) {
   return strlen($s1) - strlen($s2);
});
# print
var_dump($a1, $a2, $a3, $a4);
