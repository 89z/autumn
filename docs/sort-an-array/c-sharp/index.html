Sort an array / C#

Tue Sep 22 2020 array 
using System;
using System.Collections;

class Program {
   static void Main() {
      string[] a = {"May", "June"};
      Array.Sort(a);
      foreach (var s in a) {
         Console.WriteLine(s);
      }
   }
}
