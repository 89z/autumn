Extract string by pattern / JavaScript

Thu Jan 21 2021 string 
let s = 'Sunday Monday';
{ // example 1
   let a = s.match(/..n/);
   console.log(a[0] == 'Sun');
}
{ // example 2
   let a = s.match(/..n/g);
   console.log(a[0] == 'Sun', a[1] == 'Mon');
}
{ // example 3
   let a = s.match(/(..)n/);
   console.log(a[1] == 'Su');
}

let s = 'Sunday Monday';
{ // example 1
   let r = /..n/g;
   let a = [...s.matchAll(r)];
   console.log(a[0][0] == 'Sun', a[1][0] == 'Mon');
}
{ // example 2
   let r = /(..)n/g;
   let a = [...s.matchAll(r)];
   console.log(a[0][1] == 'Su', a[1][1] == 'Mo');
}
