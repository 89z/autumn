Go

package main

import (
   "fmt"
   "regexp"
)

func main() {
   o := regexp.MustCompile("a.")
   // example 1
   s1 := o.FindString("January")
   // example 2
   a2 := o.FindAllString("January", -1)
   // print
   fmt.Println(s1, a2)
}

package main

import (
   "fmt"
   "regexp"
)

func main() {
   o := regexp.MustCompile("a(..)")
   // example 1
   a1 := o.FindStringSubmatch("January")
   // example 2
   a2 := o.FindAllStringSubmatch("January", -1)
   // print
   fmt.Println(a1, a2)
}

Rust

use regex::{Error, Regex};

fn main() -> Result<(), Error> {
   let o_re = Regex::new("a(..)")?;
   // example 1
   let o_cap = o_re.captures("January");
   println!("{:?}", o_cap);
   // example 2
   for o_cap in o_re.captures_iter("January") {
      println!("{:?}", o_cap);
   }
   // return
   Ok(())
}
