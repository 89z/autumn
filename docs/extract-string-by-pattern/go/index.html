Extract string by pattern / Go

Wed Aug 26 2020 string 
package main

import (
   "fmt"
   "regexp"
)

func main() {
   a := regexp.MustCompile("a.").FindAllString("January", -1)
   fmt.Println(a)
}

package main

import (
   "fmt"
   "regexp"
)

func main() {
   a := regexp.MustCompile("a(..)").FindAllStringSubmatch("January", -1)
   fmt.Println(a)
}
