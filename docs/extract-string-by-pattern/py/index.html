Extract string by pattern / Python

Sat Nov 28 2020 string 
import re
s = 'Sunday Monday'
# example 1
a = re.findall('..n', s)
print(a[0] == 'Sun', a[1] == 'Mon')
# example 2
a = re.findall('(..)n', s)
print(a[0] == 'Su', a[1] == 'Mo')

https://docs.python.org/library/re.html#re.findall

import re
s = 'Sunday Monday'
# example 1
s1 = re.search('..n', s).group()
print(s1 == 'Sun')
# example 2
a = re.search('(..)n', s).groups()
print(a[0] == 'Su')
