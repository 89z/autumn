Extract string by pattern / Python

Thu May 6 2021 string 
import re
s = 'south north'
# example 1
a = re.findall('o..', s)
print(a[0] == 'out', a[1] == 'ort')
# example 2
a = re.findall('o(..)', s)
print(a[0] == 'ut', a[1] == 'rt')

https://docs.python.org/library/re.html#re.findall

import re
s = 'south north'

# example 1
t = re.search('o..', s).group()
print(t == 'out')

# example 2
a = re.search('o(..)', s).groups()
print(a[0] == 'ut')
