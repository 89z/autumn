Extract string by pattern / Rust

Sat Nov 28 2020 string 
use regex::Regex;

fn main() -> Result<(), String> {
   let o = Regex::new("(..)n").map_err(|e|
      e.to_string()
   )?;
   let a = o.captures("Sunday Monday").ok_or("captures")?;
   println!("{}", &a[1]);
   Ok(())
}

use regex::Regex;

fn main() -> Result<(), regex::Error> {
   let r = Regex::new("a(..)")?;
   for o in r.captures_iter("January") {
      println!("{:?}", o);
   }
   Ok(())
}
