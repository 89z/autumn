Number to date object

Wed Jun 10 2020 time

Go

package main
import (
   "fmt"
   "time"
)
func main() {
   var n int64 = 365 * 24 * 60 * 60
   var o = time.Unix(n, 0)
   fmt.Println(o)
}

https://golang.org/pkg/time#Unix

JavaScript

let n = 365 * 24 * 60 * 60;
let o = new Date(n * 1000);
console.log(o);

https://developer.mozilla.org/Web/JavaScript/Reference/Global_Objects/Date/Date

PHP

<?php
$n = 365 * 24 * 60 * 60;
# example 1
$o1 = new DateTime;
$o1->setTimestamp($n);
# example 2
$o2 = date_create();
date_timestamp_set($o2, $n);
# print
var_dump($o1, $o2);

https://php.net/function.date-timestamp-set

Python

n = 365 * 24 * 60 * 60
# example 1
from datetime import datetime
o1 = datetime.fromtimestamp(n)
# exmaple 2
from datetime import date
o2 = date.fromtimestamp(n)
# print
print(o1, o2)

  • https://docs.python.org/library/datetime.html#datetime.datetime.fromtimestamp
