Split a path / Nim

Sat May 8 2021 path 
import os
var s = r"C:\Windows\notepad.exe"
s = s.extractFilename
echo s == "notepad.exe"

https://nim-lang.org/docs/os.html#extractFilename,string

import os
var s = r"C:\Windows\notepad.exe"
s = s.lastPathPart
echo s == "notepad.exe"

https://nim-lang.org/docs/os.html#lastPathPart,string

import os
let s = r"C:\Windows\notepad.exe"
let (t, u, v) = s.splitFile
echo t == r"C:\Windows" and u == "notepad" and v == ".exe"

https://nim-lang.org/docs/os.html#splitFile,string
