use std::path::Path;

   let p = Path::new("a.rs");
   let s = p.extension().unwrap_or_default();
   println!("{}", s == "rs");
}

use std::path::Path;

fn main() {
   let p = Path::new(r"C:\Windows\notepad.exe");
   let q = match p.parent() {
      Some(v) => v,
      None => Path::new(r"C:\")
   };
   println!("{:?}", q);
}
