Sat May 8 2021 path 
from os import path
s = r'C:\Windows\notepad.exe'
s = path.basename(s)
print(s == 'notepad.exe')

from os import path
s = r'C:\Windows\notepad.exe'
s = path.dirname(s)
print(s == r'C:\Windows')

https://docs.python.org/library/os.path.html#os.path.dirname

from pathlib import Path
p = Path(r'C:\Windows\notepad.exe')
print(p.name == 'notepad.exe')

https://docs.python.org/library/pathlib.html#pathlib.PurePath.name

from pathlib import Path
p = Path(r'C:\Windows\notepad.exe')
s = str(p.parent)
print(s == r'C:\Windows')
