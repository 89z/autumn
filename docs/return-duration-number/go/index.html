package main

import (
   "fmt"
   "log"
)

func check(n float64, e error) {
   if e != nil {
      log.Fatal(e)
   }
   fmt.Println(n)
}

func main() {
   n, e := TimeHours("2020-05-04T03:02:01")
   check(n, e)
   n, e = TimeHours("2020-05-04T03:02")
   check(n, e)
   n, e = TimeHours("2020-05-04T03")
   check(n, e)
   n, e = TimeHours("2020-05-04")
   check(n, e)
   n, e = TimeHours("2020-05")
   check(n, e)
   n, e = TimeHours("2020")
   check(n, e)
   n, e = TimeHours("")
   check(n, e)
}

package main
import "time"

func TimeHours(left string) (float64, error) {
   right := "1970-01-01T00:00:00Z"[len(left):]
   o, e := time.Parse(time.RFC3339, left + right)
   if e != nil {
      return 0, e
   }
   return time.Since(o).Hours(), nil
}

package main
import "time"

func TimeHours(left string) (float64, error) {
   right := "1970-01-01T00:00:00Z"[len(left):]
   o, e := time.Parse(time.RFC3339, left + right)
   if e != nil {
      return 0, e
   }
   return time.Now().Sub(o).Hours(), nil
}
