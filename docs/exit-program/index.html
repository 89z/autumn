Exit program

Tue Jun 9 2020 system

D

import std.stdio;
int main() {
   "Sunday".writeln;
   return 1;
   "Monday".writeln;
}

https://dlang.org/spec/statement.html#return-statement

Go

package main
import "os"
func main() {
   println("Sunday")
   os.Exit(1)
   println("Monday")
}

package main
import "log"
func main() {
   log.Fatal("Sunday")
   println("Monday")
}

package main
func main() {
   println("Sunday")
   return
   println("Monday")
}

  • https://golang.org/pkg/log#Fatal
  • https://golang.org/pkg/os#Exit
  • https://golang.org/ref/spec#Return_statements

Nim

echo "Sunday"
if true:
   quit(QuitFailure)
echo "Monday"

PHP

<?php
echo "Sunday\n";
die(1);

<?php
echo "Sunday\n";

Python

import os
print('Sunday')
os.abort()
print('Monday')

import sys
print('Sunday')
sys.exit(1)
print('Monday')

import sys
sys.exit('Sunday')
print('Monday')

  • https://docs.python.org/library/os.html#os.abort

   println!("Monday");
}
