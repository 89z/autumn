use std::num;

fn main() -> Result<(), num::ParseIntError> {
   let n: u8 = "AB".parse()?;
   println!("{}", n);
   Ok(())
}

fn main() -> Result<(), String> {
   let n = "AB".parse::<u8>().map_err(|e|
      e.to_string()
   )?;
   println!("{}", n);
   Ok(())
}

fn main() -> Result<(), u8> {
   let n = "AB".parse::<u8>().map_err(|e| {
      println!("{}", e);
      1
   })?;
   println!("{}", n);
   Ok(())
}

use std::process;

fn main() {
   let n: u8 = match "AB".parse() {
      Ok(v) => v,
      Err(e) => {
         println!("{}", e);
         process::exit(1);
      }
   };
   println!("{}", n);
}
