package main
func main() {
   t1 := map[string]bool{"Sun": true, "Mon": true}

   _, b1 := t1["Mon"]
   // example 2

   println(b1, b2)

JavaScript

// example 1
let a1 = ['Sun', 'Mon'];
let t1 = new Set(a1);
let b1 = t1.has('Mon');

let t2 = {Sun: true, Mon: true};
let b2 = 'Mon' in t2;
// print
console.log(b1, b2);
