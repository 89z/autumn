Environment variables

Thu May 21 2020 system

Go

package main
import "os"
func main() {
   // example 1
   s1 := os.Getenv("BROWSER")
   // example 2
   s2, b1 := os.LookupEnv("BROWSER")
   // print
   println(s1, s2, b1)
}

Rust

use std::env;
use std::error::Error;
fn main() -> Result<(), Box<dyn Error>> {
   let s1 = env::var("BROWSER")?;
   dbg!(s1);
   Ok(())
}
