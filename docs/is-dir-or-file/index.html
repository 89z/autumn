Is directory or file

Fri Jun 12 2020 system

D

import std.file, std.stdio;
void main() {
   auto b = "index.md".exists;
   b.writeln;
}

https://dlang.org/library/std/file/exists.html

Dart

import 'dart:io';
main() {
   var o = new File('index.md');
   var b = o.existsSync();
   print(b);
}

https://api.dart.dev/dart-io/FileSystemEntity/existsSync.html

Go

package main
import (
   "log"
   "os"
)
func main() {
   o, e := os.Stat("index.md")
   if e != nil {
      log.Fatal(e)
   }
   log.Print(o)
}

package main
import (
   "log"
   "os"
)
func main() {
   o, e := os.Stat("index.md")
   if os.IsNotExist(e) {
      log.Fatal(e)
   }
   log.Print(o)
}

  • https://golang.org/pkg/os#IsNotExist
  • https://golang.org/pkg/os#Stat

PHP

<?php

# example 3
$b3 = is_file($s1);
# print

Python

from os import path
s1 = 'a.txt'
# example 1
b1 = path.isdir(s1)
# example 2
b2 = path.exists(s1)
# example 3
b3 = path.isfile(s1)
# print
print(b1, b2, b3)
